\begin{gather\*}
  \sum\_{i=1}^{n} c = cn\\\\
  \sum\_{i=1}^{n} i = \frac{n(n+1)}{2}\\\\
  \sum\_{n=0}^{N-1} r^n = \frac{1-r^N}{1-r}
\end{gather\*}
