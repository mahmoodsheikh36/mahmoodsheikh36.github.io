find the following integrals

\[ \int e^\frac{x}{2} \sin(7x) \text{ dx} \]

\begin{align*} \int e^\frac{x}{2} \sin(7x) \text{ dx} &= 2e^{\frac{x}{2}}\sin(7x) - \int 14e^{\frac{x}{2}}\cos(7x) \text{ dx}\\ &= 2e^{\frac{x}{2}}\sin(7x) - 14\int e^{\frac{x}{2}}\cos(7x) \text{ dx}\\ &= 2e^{\frac{x}{2}}\sin(7x) - 14\left(2e^{\frac{x}{2}}\cos(7x) - \int 2e^{\frac{x}{2}} \cdot -7\sin(7x) \text{ dx}\right)\\ &= 2e^{\frac{x}{2}}\sin(7x) - 28e^{\frac{x}{2}}\cos(7x) - 196\int e^{\frac{x}{2}}\sin(7x) \text{ dx}\\ &= \frac{2e^{\frac{x}{2}}\sin(7x)}{197} - \frac{28e^{\frac{x}{2}}\cos(7x)}{197} \end{align*}

\[ \int \sin^9\left(\frac{x}{3}\right) \cos^5\left(\frac{x}{3}\right) \text{ dx} \]

\begin{align*} t=\sin \left(\frac{x}{3}\right),\ \text{dt}&=\frac{1}{3}\cos\left(\frac{x}{3}\right) \text{ dx},\ \text{dx} = \frac{3\text{ dt}}{\cos\frac{x}{3}}\\ \int \sin^9\left(\frac{x}{3}\right) \cos^5\left(\frac{x}{3}\right) \text{ dx} &= \int t^9\cos^5\left(\frac{x}{3}\right) \frac{3\text{ dt}}{\cos\frac{x}{3}}\\ &= 3\int t^9\cos^4\left(\frac{x}{3}\right) \text{ dt}\\ &= 3\int t^9\left(1-\sin^2\left(\frac{x}{3}\right)\right)^2 \text{ dt}\\ &= 3\int t^9\left(1-t^2\right)^2 \text{ dt}\\ &= 3\int t^9\left(1-2t^2+t^4\right) \text{ dt}\\ &= 3\int t^9-2t^{11}+t^{13} \text{ dt}\\ &= 3\left(\frac{t^{10}}{10} - \frac{2t^{12}}{12} + \frac{t^{14}}{14}\right)\\ &= \frac{3t^{10}}{10} - \frac{t^{12}}{2} + \frac{3t^{14}}{14}\\ &= \frac{3\sin^{10}\left(\frac{x}{3}\right)}{10} - \frac{\sin^{12}\left(\frac{x}{3}\right)}{2} + \frac{3\sin^{14}\left(\frac{x}{3}\right)}{14} \end{align*}

\[ \int \frac{5}{8\cos x+10-10\sin x} \text{ dx} \]

\begin{align*} \int \frac{5}{8\cos x+10-10\sin x} \text{ dx} &= \int \frac{5}{8\cdot\frac{1-t^2}{1+t^2}+10-10\cdot\frac{2t}{1+t^2}} \frac{2\text{ dt}}{1+t^2}\\ &= \int \frac{10}{8\cdot(1-t^2)+10(1+t^2)-10\cdot2t} \text{ dt}\\ &= 5\int \frac{1}{9+t^2-10t} \text{ dt}\\ &= 5\int \frac{A}{t-9} + \frac{B}{t-1} \text{ dt}\\ \end{align*}

we solve for \(A\) and \(B\)

\begin{gather*} \frac{1}{9+t^2-10t} = \frac{A}{t-9} + \frac{B}{t-1}\\ 1 = A(t-1)+B(t-9) \implies 1 = t(A+B)-(A+9B)\\ \implies A+B=0,\ -A-9B=1 \implies B=\frac{1}{8},A=-\frac{1}{8}\\ \end{gather*}

we return to the original integration:

\begin{align*} \int \frac{5}{8\cos x+10-10\sin x} \text{ dx} &= 5\int \frac{-1}{8(t-9)} + \frac{1}{8(t-1)} \text{ dt}\\ &= \frac{5}{8}\int \frac{1}{t-1} - \frac{1}{t-9} \text{ dt}\\ &= \frac{5}{8} \ln(t-1)- \frac{5}{8}\ln(t-9)\\ &= \frac{5}{8} \ln\left(\tan\frac{x}{2}-1\right)- \frac{5}{8}\ln\left(\tan\frac{x}{2}-9\right) \end{align*}

\[ \int \frac{6x^2+10x+8}{(x^2-1)(x+2)} \text{ dx} \]

\begin{align*} \int \frac{6x^2+10x+8}{(x^2-1)(x+2)} \dif{x} &= \int \frac{6x^2+10x+8}{(x-1)(x+1)(x+2)} \dif{x}\\ &= \int \frac{A}{x-1} + \frac{B}{x+1} + \frac{C}{x+2} \dif{x} \end{align*}

we solve for \(A,B,C\)

\begin{gather*} \frac{6x^2+10x+8}{(x-1)(x+1)(x+2)} = \frac{A}{x-1} + \frac{B}{x+1} + \frac{C}{x+2}\\ 6x^2+10x+8 = A(x+1)(x+2) + B(x-1)(x+2) + C(x+1)(x-1)\\ \vdots\\ \frac{4}{x+2}-\frac{2}{x+1}+\frac{4}{x-1} \end{gather*}

we return to the original integration

\begin{align*} \int \frac{6x^2+10x+8}{(x^2-1)(x+2)} \dif{x} &= \int \frac{4}{x-1} + \frac{2}{x+1} + \frac{4}{x+2} \dif{x}\\ &= 4\ln(x-1) + 2\ln(x+1) + 4\ln(x+2) \end{align*}

find the following limit
\[ \lim_{n\to\infty} \left(\frac{1^2}{n^3+1^3}+\frac{2^2}{n^3+2^3}+\frac{3^2}{n^3+3^3}+\cdots+\frac{n^2}{n^3+n^3}\right) \]

this limit represents a riemann sum

\begin{align*} \lim_{n\to\infty} \left(\frac{1^2}{n^3+1^3}+\frac{2^2}{n^3+2^3}+\cdots+\frac{n^2}{n^3+n^3}\right) &= \lim_{n\to\infty} \left(\frac{\frac{1^2}{n^3}}{1+\frac{1^3}{n^3}}+\frac{\frac{2^2}{n^3}}{1+\frac{2^3}{n^3}}+\cdots+\frac{\frac{n^2}{n^3}}{1+\frac{n^3}{n^3}}\right)\\ &= \lim_{n\to\infty} \frac{1}{n}\left(\frac{\frac{1^2}{n^2}}{1+\frac{1^3}{n^3}}+\frac{\frac{2^2}{n^2}}{1+\frac{2^3}{n^3}}+\cdots+\frac{\frac{n^2}{n^2}}{1+\frac{n^3}{n^3}}\right)\\ &= \lim_{n\to\infty} \frac{1}{n}\left(\frac{\left(\frac{1}{n}\right)^2}{1+\left(\frac{1}{n}\right)^3}+\frac{\left(\frac{2}{n}\right)^2}{1+\left(\frac{2}{n}\right)^3}+\cdots+\frac{\left(\frac{n}{n}\right)^2}{1+\left(\frac{n}{n}\right)^3}\right)\\ &= \int_{0}^{1} \frac{x^2}{1+x^3} \dif{x}\\ &= \frac{1}{3}\int_{0}^{1} \frac{3x^2}{1+x^3} \dif{x}\\ &= \frac{1}{3} \left. \ln(1+x^3) \right|_0^1\\ &= \frac{1}{3} \ln2 \end{align*}
