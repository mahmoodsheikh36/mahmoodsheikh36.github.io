2 write struct to file

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct person
{
  int id;
  char fname[20];
  char lname[20];
};

int main()
{
  FILE *outfile;

  // open file for writing
  outfile = fopen("person.dat", "w");
  if (outfile == NULL)
  {
    fprintf(stderr, "\nError opened file\n");
    exit(1);
  }

  struct person input1 = {1, "rohan", "sharma"};
  struct person input2 = {2, "mahendra", "dhoni"};

  // write struct to file
  fwrite(&input1, sizeof(struct person), 1, outfile);
  fwrite(&input2, sizeof(struct person), 1, outfile);

  if(fwrite != 0)
    printf("contents written to file\n");
  else
    printf("error writing to file\n");

  // close file
  fclose(outfile);

  return 0;
}

3 read struct from file

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <fcntl.h>

struct phone {
  char first_name[20];
  char last_name[20];
  char phone_num[15];
};

const unsigned int PHONE_SIZE = sizeof(struct phone);

void print_struct(int infile) {
  struct phone input;
  read(infile, &input, PHONE_SIZE);
  printf("First name: %s\nLast name: %s\nPhone number: %s\n",
         input.first_name, input.last_name, input.phone_num);
}

int main() {
  int infile;
  infile = open("phones.dat", O_RDONLY);
  if (infile == -1) {
    perror("\nerror opening file\n");
    exit(1);
  }

  int filesize = lseek(infile, 0, SEEK_END);

  int pos = lseek(infile, 0, SEEK_SET);
  while (pos < filesize) {
    print_struct(infile);
    pos = lseek(infile, pos+PHONE_SIZE, SEEK_SET);
  }

  close(infile);

  return 0;
}
