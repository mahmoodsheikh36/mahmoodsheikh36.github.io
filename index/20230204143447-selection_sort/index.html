selection sort

selection sort maintains two subarrays in a given array, one subarray that is sorted and the other unsorted, it sorts a given array by repeatedly finding the minimum element (considering ascending order) from the unsorted part and putting it at the beginning in every iteration of selection sort, the minimum element (considering ascending order) from the unsorted subarray is picked and moved to the sorted subarray. time complexity is 

template <typename T>
void selection_sort(T arr[], int len) {
  for (int i = 0; i < len; ++i) {
    // find index of minimum in unsorted subarray
    int min_idx = i;
    for (int j = i+1; j < len; ++j) {
      if (arr[j] < arr[min_idx]) {
        min_idx = j;
      }
    }

    // add minimum to the sorted subarray
    T tmp = arr[i];
    arr[i] = arr[min_idx];
    arr[min_idx] = tmp;
  }
}

example usage:

#include <iostream>
<<SelectionSort>>
int main() {
  int arr[] = {8, 1, 2, 10, 50, 18};
  selection_sort(arr, std::size(arr));
  for (int i = 0; i < std::size(arr); ++i) {
    std::cout << arr[i] << std::endl;
  }
}
