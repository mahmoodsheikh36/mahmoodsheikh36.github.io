def encrypt(plaintext, key):
    """scytale cipher encryptor, key should be an integer"""
    portion_length = len(plaintext) // key
    if len(plaintext) % key > 0:
        portion_length += 1

    # extract the portions (horizontal lines)
    portions = []
    for i in range(key):
        portion_start_idx = i * portion_length
        portion_end_idx = portion_start_idx + portion_length
        portions.append(plaintext[portion_start_idx:portion_end_idx])

    # iterate through the portions, extracting the final line
    # line is initially a list of strings, to avoid string concatenation on each iteration, think java's stringbuilders
    line = []
    for i in range(portion_length):
        for j in range(key):
            portion = portions[j]
            if len(portion) > i:
                mychar = portion[i]
                line.append(mychar)
    return ''.join(line)

this function runs in time linear to the plaintext length (dont let the nested loop fool you into thinking its quadratic!). example usage of the function:

def decrypt(ciphertext, key):
    portion_length = len(ciphertext) // key
    if len(ciphertext) % key > 0:
        portion_length += 1
    portions = [[] for _ in range(key)]
    gap = portion_length * key - len(ciphertext)
    rect_i = 0
    i = 0
    while i < len(ciphertext):
        if (rect_i + 1) % key == 0 and (rect_i + 1) // key > (portion_length - gap) and\
           rect_i < len(ciphertext):
            pass
        else:
            portions[rect_i % key].append(ciphertext[i])
            i += 1
        rect_i += 1
    return ''.join([''.join(portion) for portion in portions])

breaking the cipher

automatic letter-frequency analysis combined with key space bruteforce could solve this, i mean iterating through all possible keys upto the length of the text and for each key checking if the frequency of letters decrypting with it produces to see if they match the traditional letter-frequency in english texts, but that wouldnt work for small texts and isnt guaranteed to work anyway the key space could be reduced to numbers that divide the text length evenly to simplify the problem 🤔 but im not sure im even allowed to do that the idea i eventually came up with is iterating through every possible key, decrypting the text using the key and comparing the distribution of bigrams in the received text to the average distribution of bigrams in english text, the key that results in the closest distribution to the average distribution would be the most likely one. for a given ciphertext , let be an arbitrary key, would be the message received by decrypting using the key , each bigram (pair) would be a random variable drawn from the distribution given in tab-eng-bigram-freq, we assume these random variables are independent and identically distributed. tab-eng-bigram-freq doesnt provide probabilities (or frequencies) for all possible pairs of letters, as the total possibilities is . if the pair isnt in the table, we consider its frequency to be equal to the lowest frequency in the table. here, we use the technique of maximum likelihood estimation, we want to find the estimate : where is our key space. because we're assuming the random variables are i.d.d, the their joint probability is simply the product of their probabilities, and since the key space is finite and is bounded by the size of the ciphertext, we could just iterate through each possible key to find the one that produces the maximum likelihood.

import re

FREQS_MAP = {entry[0]: entry[1] for entry in FREQS}

def min_probability():
    mymin = 100
    for freq in FREQS_MAP.values():
        if freq < mymin:
            mymin = freq
    return mymin / 100

V_MIN_PROBABILITY = min_probability()

def likelihood(text):
    idx = 0
    v_likelihood = 1
    while idx < len(text):
        bigram = text[idx:idx+2]
        if bigram in FREQS_MAP:
            v_likelihood *= FREQS_MAP[bigram]
        else:
            v_likelihood *= V_MIN_PROBABILITY
        idx += 1
    return v_likelihood

def break_cipher(ciphertext):
    min_l = -1
    min_l_key = None
    for key in range(1, len(ciphertext) + 1):
        new_l = likelihood(decrypt(ciphertext, key))
        if new_l > min_l:
            min_l = new_l
            min_l_key = key
    return min_l_key
