\begin{definition}
a \textbf{directed graph} is a \href{20230314001051-graph.org}{graph} whose \href{20230314001632-edge.org}{edge}s are all directed\\
\end{definition}
\begin{definition}
in a directed graph, the \textbf{outdegree} of a vertex is the number of the edges whose tail is connected to the vertex and the \textbf{indegree} is the number of edges whose head is connected to the vertex\\
\end{definition}
