\begin{tabular}{ c|c|c|c|c }
  & ab=00 & ab=01 & ab=11 & ab=10\\
  \hline
  cd=00 & 1 & 1 & 0 & 1\\
  cd=01 & 0 & 1 & 1 & 1\\
  cd=11 & 1 & 1 & 1 & 0\\
  cd=10 & 1 & 0 & 1 & 1
\end{tabular}
