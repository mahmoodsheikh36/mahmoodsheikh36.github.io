(defun tree-to-latex-forest-for-org (tree)
  "turn a tree into a runnable org mode latex block"
  (format t "#+begin_src latex :results file graphics :file (temp-file \"svg\") ~%\\begin{forest}")
  (tree-to-latex-forest-recurse tree)
  (format t "\\end{forest}~%#+end_src"))

(defun tree-to-latex-forest-recurse (tree)
  "helper function for tree-to-latex-forest-for-org"
  (if (not (eq tree nil))
      (progn
        (format t "[~a,circle,draw" (car tree))
        (loop for child in (cdr tree)
              do (tree-to-latex-forest-recurse child))
        (format t "]"))
      (format t "[,phantom]"))
  tree)

(let ((my-tree '(23 (12) (12 (1 nil (4 nil (5 nil (9 nil (10 nil nil))))) nil) (28 nil (45 nil (89 nil nil))))))
  (tree-to-latex-forest-for-org my-tree))
