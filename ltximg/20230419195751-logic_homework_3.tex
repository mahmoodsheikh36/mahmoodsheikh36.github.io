\begin{question}
write the following expression in \href{20230422185928-dnf.org}{DNF} and \href{20230422185859-cnf.org}{CNF} forms\\
\begin{subquestion}
\((p \rightarrow (q \rightarrow r)) \rightarrow ((p \rightarrow q) \rightarrow (p \rightarrow r))\)\\
\begin{answer}
in dnf form:\\
\begin{align*}
  (p \rightarrow (q \rightarrow r)) \rightarrow ((p \rightarrow q) \rightarrow (p \rightarrow r)) &\equiv \lnot(\lnot p \lor (\lnot q \lor r)) \lor (\lnot(\lnot p \lor q) \lor (\lnot p \lor r))\\
  &\equiv \lnot(\lnot p \lor \lnot q \lor r) \lor \lnot(\lnot p \lor q) \lor (\lnot p \lor r)\\
  &\equiv \lnot(\lnot p \lor \lnot q \lor r) \lor \lnot(\lnot p \lor q) \lor \lnot p \lor r\\
  &\equiv \lnot(\lnot p \lor \lnot q \lor r) \lor (p \land \lnot q) \lor \lnot p \lor r\\
  &\equiv (p \land q \land \lnot r) \lor (p \land \lnot q) \lor \lnot p \lor r\\
\end{align*}
in cnf form:\\
\begin{align*}
  (p \rightarrow (q \rightarrow r)) \rightarrow ((p \rightarrow q) \rightarrow (p \rightarrow r)) &\equiv (p \land q \land \lnot r) \lor (p \land \lnot q) \lor \lnot p \lor r\\
  &\equiv (q \land \lnot r) \lor \lnot q \lor \lnot p \lor r\\
  &\equiv q \lor \lnot q \lor \lnot p \lor r\\
  &\equiv T
\end{align*}
\end{answer}
\end{subquestion}
\begin{subquestion}
\((p \rightarrow q) \rightarrow ((p \land \lnot q) \rightarrow r)\)\\
\begin{answer}
in dnf form:\\
\begin{align*}
  (p \rightarrow q) \rightarrow ((p \land \lnot q) \rightarrow r) &\equiv \lnot(\lnot p \lor q) \lor (\lnot(p \land \lnot q) \lor r)\\
  &\equiv (p \land \lnot q) \lor (\lnot(p \land \lnot q) \lor r)\\
  &\equiv (p \land \lnot q) \lor ((\lnot p \lor q) \lor r)\\
  &\equiv (p \land \lnot q) \lor (\lnot p \lor q \lor r)\\
  &\equiv (p \land \lnot q) \lor \lnot p \lor q \lor r
\end{align*}
in cnf form:\\
\end{answer}
\begin{align*}
  (p \rightarrow q) \rightarrow ((p \land \lnot q) \rightarrow r) &\equiv (p \land \lnot q) \lor \lnot p \lor q \lor r\\
  &\equiv (p \land \lnot q) \lor \lnot p \lor q \lor r\\
  &\equiv (p \land \lnot q) \lor r\\
  &\equiv (p \lor r) \land (\lnot q \lor r)
\end{align*}
\end{subquestion}
\begin{subquestion}
\(((p \rightarrow q) \rightarrow p) \rightarrow \lnot p\)\\
\begin{answer}
in dnf form:\\
\begin{align*}
  ((p \rightarrow q) \rightarrow p) \rightarrow \lnot p &\equiv \lnot(\lnot(\lnot p \lor q) \lor p) \lor \lnot p\\
  &\equiv ((\lnot p \lor q) \land \lnot p) \lor \lnot p\\
  &\equiv ((\lnot p \land \lnot p) \lor (q \land \lnot p)) \lor \lnot p\\
  &\equiv (q \land \lnot p) \lor \lnot p
\end{align*}
in cnf form:\\
\begin{align*}
  ((p \rightarrow q) \rightarrow p) \rightarrow \lnot p &\equiv (q \land \lnot p) \lor \lnot p\\
  &\equiv q \land \lnot p
\end{align*}
\end{answer}
\end{subquestion}
\begin{subquestion}
\(((p \rightarrow q) \land (q \rightarrow \lnot r)) \rightarrow (p \rightarrow r)\)\\
\begin{answer}
in dnf form:\\
\begin{align*}
  ((p \rightarrow q) \land (q \rightarrow \lnot r)) \rightarrow (p \rightarrow r) &= \lnot((\lnot p \lor q) \land (\lnot q \lor \lnot r)) \lor (\lnot p \lor r)\\
  &= (\lnot(\lnot p \lor q) \lor \lnot(\lnot q \lor \lnot r)) \lor \lnot p \lor r\\
  &= ((p \land \lnot q) \lor (q \land r)) \lor \lnot p \lor r\\
  &= (p \land \lnot q) \lor (q \land r) \lor \lnot p \lor r
\end{align*}
in cnf form:\\
\begin{align*}
  ((p \rightarrow q) \land (q \rightarrow \lnot r)) \rightarrow (p \rightarrow r) &= (p \land \lnot q) \lor (q \land r) \lor \lnot p \lor r\\
  &= \lnot q \lor (q \land r) \lor \lnot p \lor r\\
  &= \lnot q \lor r \lor \lnot p\\
  &= (\lnot q \lor r \lor \lnot p) \land (\lnot q \lor r \lor \lnot p)\\
\end{align*}
\end{answer}
\end{subquestion}

\begin{subquestion}
write an expression with the variables \(p,q,r,s\) that receives T iff most of the variables have the value T\\
\begin{answer}
\[
(p \land q \land r) \lor (p \land q \land s) \lor (p \land r \land s) \lor (q \land r \land s)
\]\\
\end{answer}

\begin{question}
write an expression with the variables \(p,q,r\) that receives T iff most of its variables have the value T\\
\begin{answer}
\[
(p \land q) \lor (p \land r) \lor (q \land r)
\]\\
\end{answer}
