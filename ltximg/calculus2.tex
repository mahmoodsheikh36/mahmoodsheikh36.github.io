\section{limit}
\label{sec:org8cc3125}
\begin{definition}
the \textbf{limit} of a sequence is denoted as\\[0pt]
\[
  \lim_{n\to\infty} \frac1n = L
\]\\[0pt]
if for every \(\epsilon > 0\) there exists \(0 < N,\ N(\epsilon)\) such that for every \(n > N\) we know \(|a_n-L| < \epsilon\)\\[0pt]

\begin{my_example}
\[
  \lim_{n\to\infty} \frac{2n+4}{7n+5}
\]\\[0pt]
to find this limit we start with \(|a_n-L|\) which we need to show is smaller than \(\epsilon\):\\[0pt]
\[ |a_n - L| = \left|\frac{2n+4}{7n+5} - \frac27\right| = \left|\frac{7(2n+4)-2(7n+5)}{7(7n+5)}\right| = \frac{18}{49n+35} < \frac{18}{49n} < \epsilon \implies \frac{18}{49\epsilon} < n \implies N(\epsilon) = \left[\frac{18}{49\epsilon}\right]+1 \]\\[0pt]
\end{my_example}
\end{definition}

\subsection{Squeeze theorem}
\label{sec:org6d719ce}
the \textbf{squeeze theorem} states that if \(f(x) \le g(x) \le h(x)\) for all numbers, and at some point \(x=k\) we have \(f(k)=h(k)\), then \(g(k)\) must also be equal to them.\\[0pt]
we can use the theorem to find tricky limits like \(\lim_{x \to 0} \frac{\sin x}{x}\), by "squeezing" \(\frac{\sin x}{x}\) between two nicer functions and using them to find the limit\\[0pt]

\subsection{limits using euler}
\label{sec:org5af9ed3}
\begin{note}
need to add citation to \url{https://marcocetica.com/posts/euler\_number\_calculus/} (took lots of stuff from here!)\\[0pt]
\end{note}
a consequence of the definition of \href{20220704213259-euler_s_number.org}{Euler's Number} is:\\[0pt]
\[
  \lim_{x\to \pm\infty} \left(1+\frac{1}{f(x)}\right)^{f(x)} = e \iff f(x) \to \pm\infty
\]\\[0pt]
thus, if we have an \hyperref[sec:orgd592a44]{indeterminate form} that can be rewritten in this form, it would be easy to calculate it\\[0pt]

\begin{my_example}
\[
  \lim_{x\to \infty} \left(1+\frac{3}{x}\right)^x
\]\\[0pt]
substitution of \(x=\infty\) give us \(1^\infty\) which is indeterminate, but since \(f(x)=\frac{x}{3}\to\infty\) we can use \href{20220704213259-euler_s_number.org}{Euler's Number} to get its value\\[0pt]
we rewrite the limit as:\\[0pt]
\[
  \lim_{x\to\infty} \left(1+\frac{1}{\frac{x}{3}}\right)^x
\]\\[0pt]
now we need to move the 3 to the exponent, to do that we can multiply and divide the exponent by 3\\[0pt]
\[
  \lim_{x\to\infty} \left(1+\frac{1}{\frac{x}{3}}\right)^{x\cdot\frac{3}{3}} = \lim_{x\to\infty} \left(\underbrace{\left(1+\frac{1}{\frac{x}{3}}\right)^{\frac{x}{3}}}_{e}\right)^3 = e^3
\]\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \lim_{x \to \infty} \left(  \frac{3x+1}{3x-5} \right)^{x-2}
\]\\[0pt]
again, this is an indeterminate form \(1^\infty\). the first thing we can do to obtain the standard form is to add and subtract a quantity to the numerator of the fraction. in this case this quantity is 5\\[0pt]
\[
  \lim_{x\to\infty} \left(\frac{3x-5+5+1}{3x-5}\right)^{x-2}
\]\\[0pt]
we can now split the fraction:\\[0pt]
\[
  \lim_{x\to\infty} \left(\frac{3x-5}{3x-5} + \frac{6}{3x-5}\right)^{x-2} = \lim_{x\to\infty} \left(1+\frac{6}{3x-5}\right)^{x-2}
\]\\[0pt]
at this point we need to "adjust" our fraction in order to have the standard form\\[0pt]
\[
  \lim_{x\to\infty} \left(1+\frac{1}{\frac{3x+5}{6}}\right)^{x-2}
\]\\[0pt]
finally, we multiply and divide the exponent by \(f(x)\)\\[0pt]
\[
  \lim_{x\to\infty} \left(1+\frac{1}{\frac{3x+5}{6}}\right)^{(x-2) \cdot \frac{3x+5}{6} \cdot \frac{6}{3x+5}}
\]\\[0pt]
which led us to:\\[0pt]
\[
  \lim_{x\to\infty} \left(\left(1+\frac{1}{\frac{3x+5}{6}}\right)^{\frac{3x+5}{6}}\right)^{\frac{6x-12}{3x+5}} = e^\frac{6}{3} = e^2
\]\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \lim_{n\to\infty} \left(\frac{n^2-5n+6}{n^2-9n+20}\right)^{2n-1}
\]\\[0pt]
in this case, we first need to divide the numerator by the denominator (i.e., \hyperref[sec:org4103a69]{polynomial long division}). This leaves us with two new polynomials:\\[0pt]
\[
  P(x)=1 \qquad R(x)=4n-14
\]\\[0pt]
we can rewrite the limit like this:\\[0pt]
\[
  \lim_{n\to\infty} \left(1+\frac{4n-14}{n^2-9n+20}\right)^{2n-1}
\]\\[0pt]
the rest is similar to the previous example\\[0pt]
\[
  = \lim_{n\to\infty} \left(1+\frac{1}{\frac{n^2-9n+20}{4n-14}}\right)^{2n-1 \cdot \frac{n^2-9n+20}{4n-14} \cdot \frac{4n-14}{n^2-9n+20}} = \lim_{n\to\infty} \left( \underbrace{\left(1+\frac{1}{\frac{n^2-9n+20}{4n-14}}\right)^{\frac{n^2-9n+20}{4n-14}}}_{e}\right)^{\frac{(4n-14)(2n-1)}{n^2-9n+20}} = e^8
\]\\[0pt]
\end{my_example}
\subsection{L'Hôpital's rule}
\label{sec:org3873de8}
see \url{https://en.wikipedia.org/wiki/L\%27H\%C3\%B4pital\%27s\_rule}\\[0pt]
\textbf{L'Hôpital's rule} states that for functions \(f\) and \(g\) which are differentiable on an open interval \(I\), except possibly at a point \(c\) contained in \(I\), if \(\lim_{x\to c} f(x) = \lim_{x\to c} g(x) = 0\) or \(\lim_{x\to c} f(x) = \lim_{x\to c} g(x) = \pm\infty\), and \(g'(x) \neq 0\) for all \(x\) in \(I\) with \(x \neq c\) and \(\lim_{x\to c} \frac{f'(x)}{g'(x)}\) exists then\\[0pt]
\[
  \lim_{x\to c} \frac{f(x)}{g(x)} = \lim_{x\to c} \frac{f'(x)}{g'(x)}
\]\\[0pt]
the differentiation of the numerator and denominator often simplifies the quotient or converts it to a limit that can be evaluated directly.\\[0pt]

\begin{my_example}
\[
  \lim_{x\to \infty} x\sin\frac1x
\]\\[0pt]
if we were to substitute \(x=\infty\) we would get:\\[0pt]
\[
   \infty \cdot \sin\frac1\infty = \infty \cdot \sin0 = \infty \cdot 0
\]\\[0pt]
which is an \hyperref[sec:orgd592a44]{indeterminate form}\\[0pt]
we can rewrite the given expression as a fraction of 2 functions:\\[0pt]
\[
  x\sin\frac1x =\frac{\sin\frac1x}{\frac1x}
\]\\[0pt]
now we can substitute \(t=\frac{1}{x}\) to make our lives easier, so we get:\\[0pt]
\[
  \lim_{x\to \infty} \frac{\sin\frac1x}{\frac1x} = \lim_{t\to 0} \frac{\sin{t}}{t}
\]\\[0pt]
notice how the limit changed from \(\lim_{x\to \infty}\) to \(\lim_{t\to 0}\), this is because we substituted \(t=\frac{1}{x}\), and \(\lim_{x\to \infty} \frac{1}{x} = 0\)\\[0pt]
now since \(\lim_{t\to 0} \sin{t} = \lim_{t\to 0} t = 0\) we can make use of \hyperref[sec:org3873de8]{L'Hôpital's rule}\\[0pt]
\[
  \lim_{t\to 0} \frac{\sin{t}}{t} = \lim_{t\to 0} \frac{\cos{t}}{1} = 1
\]\\[0pt]
and since this limit is equal to the original limit, we finally get:\\[0pt]
\[
  \lim_{x\to \infty} x\sin\frac1x = 1
\]\\[0pt]
\end{my_example}

\subsection{definite integral}
\label{sec:orgb402e53}
\begin{definition}
given a function \(f(x)\) that is continuous on the interval \([a,b]\), we divide the interval into \(n\) subintervals of equal width, \(\Delta x\), and from each interval choose a point, \(x_i^*\). then the \textbf{definite integral} of \(f(x)\) from \(a\) to \(b\) is:\\[0pt]
\[
  \int_a^b f(x) \text{ dx} = \lim_{n\rightarrow\infty} \sum_{i=1}^{n} f(x_i^*) \Delta x
\]\\[0pt]
in practicality:\\[0pt]
\[ \int f(x) \text{ dx} = F(x) + c \Rightarrow \int_a^b = F(b) - F(a) \]\\[0pt]

\begin{my_example}
\[
  \int_1^3 x^2 \text{ dx} = \frac{x^3}{3} \Big|_1^3 = \frac{3^3}{3} - \frac{1^3}{3} = 8\frac23
\]\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \int_1^3 x^2 - 4x + 3 \text{ dx} = \frac{x^3}{3} - \frac{4x^2}{2} + 3x \Big|_1^3 = \frac{3^3}{3} - \frac{4 \cdot 3^2}{2} + 3 \cdot 3 - \left(\frac{1^3}{3} - \frac{4 \cdot 1^2}{2} + 3 \cdot 1\right) = -\frac43
\]\\[0pt]
\end{my_example}

\subsection{finding area using definite integral}
\label{sec:org433c754}
\begin{definition}
the area between 2 functions \(f,g\) in an interval \([a,b]\) in which \(f \geq g\) or \(g \geq f\) is defined as:\\[0pt]
\[
  S = \left| \int_a^b (f(x) - g(x)) \text{ dx} \right|
\]\\[0pt]

\begin{note}
notice that the definition says \([a,b]\) in which \(f \geq g\) or \(g \geq f\) which means that if the functions cross each other in that interval and the lower function becomes the upper and the upper function becomes the lower then the area cant be calculated using this single formula but we split the interval into multiple and add the results\\[0pt]
\end{note}

\begin{my_example}
consider the area \(S\) in the interval \([0,3]\) between \(f(x)=x^2\) and the \(x\) axis (\(g(x)=0\))\\[0pt]

\includesvg{/Users/mahmooz/brain/out/p44azc}\\[0pt]

we calculate using the formula, i didnt mind the fact that the result should be turned to its absolute value because i know the \(f(x) \geq g(x)\)\\[0pt]

\[
  S = \int_0^3 (f(x) - g(x)) \text{ dx} = \int_0^3 x^2 \text{ dx} = \frac{x^3}{3} \Big|_0^3 = \frac{3^3}{3} - \frac{0^3}{3} = 9
\]\\[0pt]
\end{my_example}

\begin{my_example}
consider the area \(S_1 + S_2\) in red between the polynomials\\[0pt]

\includesvg{/Users/mahmooz/brain/out/PzjSrV}\\[0pt]

we split the area into 2 because if we tried to find the area without splitting we would be adding a negative value to a positive value whereas the areas should be positive, we wouldnt mind this when working with functions that dont cross because we can always take the absolute value of the result\\[0pt]

so we find \(S_1\) and \(S_2\) and the total area would be \(|S_1| + |S_2|\), assume \(a=-2,\ b=2\), \(S_1\) would be the area from \([0,2]\) and \(S_2\) the area from \([0,2]\)\\[0pt]

for \(S_1\) we take \(f(x) - g(x)\) because \(f(x) \geq g(x)\)\\[0pt]

\begin{align*}
  S_1 = \int_{-2}^0 (f(x) - g(x)) \text{ dx} &= \int_{-2}^0 3x^3-x^2-10x-(-x^2+2x) \text{ dx}\\
  &= \int_{-2}^0 3x^3-12x \text{ dx}\\
  &= \frac{3x^4}{4}-\frac{12x^2}{2} \Big|_{-2}^0\\
  &= 0 - \left(\frac{3(-2)^4}{4}-\frac{12(-2)^2}{2}\right)\\  
  &= 12
\end{align*}

for \(S_2\) we take \(f(x) - g(x)\) because \(f(x) \geq g(x)\)\\[0pt]

\begin{align*}
  \int_0^2 (g(x)-f(x)) \text{ dx} &= \int_0^2 -x^2+2x-(3x^3-x^2-10x) \text{ dx}\\
  &= \int_0^2 -x^2+2x-3x^3+x^2+10x \text{ dx}\\
  &= \int_0^2 -3x^3+12x \text{ dx}\\
  &= \left(\frac{-3x^4}{4} + \frac{12x^2}{2}\right) \Big|_0^2\\
  &= \frac{-3 \cdot 2^4}{4} + \frac{12 \cdot 2^2}{2}\\
  &= 12
\end{align*}

therefore \(S = S_1 + S_2 = 24\)\\[0pt]
\end{my_example}

\begin{note}
notice that these examples provide you with a plot of the functions so its easier to find the area, in general you wouldnt have the plot and you have to find the intersection points yourself so that you can integrate and find the area, see the following exercise for example\\[0pt]
\end{note}

\begin{my_example}
find the area between \(f(x)=x^3-4x^2+13x-7,\ g(x)=6x^2-x^3+25x-7\)\\[0pt]
first we find the intersection points so that we know how to split the integration\\[0pt]

\begin{gather*}
  x^3-4x^2+13x-7=6x^2-x^3+25x-7\\
  2x^3-10x^2-12x=0\\
  x(2x^2-10x-12)=0\\
  2x(x-6)(x+1)=0\\
  \Rightarrow x=0,6,-1
\end{gather*}

we found that we have intersections at \(x=0,6,-1\), so we find the integrals in the intervals \([-1,0],[0,6]\)\\[0pt]

\begin{align*}
  \int_{-1}^0 (f(x)-g(x)) \text{ dx} &= \int_{-1}^0 x^3-4x^2+13x-7-(6x^2-x^3+25x-7) \text{ dx}\\
  &= \int_{-1}^0 2x^3 - 10x^2 - 12x \text{ dx}\\
  &= \left(\frac{2x^4}{4} - \frac{10x^3}{3} - \frac{12x^2}{2}\right) \Big|_{-1}^0\\
  &= 0-\left(\frac{2(-1)^4}{4} - \frac{10(-1)^3}{3} - \frac{12(-1)^2}{2}\right)\\
  &= 0-\left(0.5 + \frac{10}{3} - 6\right)\\
  &= \frac{13}{6}
\end{align*}

\begin{align*}
  \int_0^6 (f(x)-g(x)) \text{ dx} &= \int_0^6 2x^3 - 10x^2 - 12x \text{ dx}\\
  &= \left(\frac{2x^4}{4} - \frac{10x^3}{3} - \frac{12x^2}{2}\right) \Big|_0^6\\
  &= \frac{2 \cdot 6^4}{4} - \frac{10 \cdot 6^3}{3} - \frac{12 \cdot 6^2}{2}\\
  &= -288
\end{align*}

area is \(288\frac{13}{6}\)\\[0pt]
\end{my_example}
\end{definition}
\subsection{solid of revolution}
\label{sec:org54164d2}
\begin{definition}
\textbf{disc integration} is a method for calculating the volume of a \textbf{solid of revolution} of a solid-state material when integrating along an axis "parallel" to the axis of revolution\\[0pt]
if the functions rotate around the \(x\) axis, the formula is\\[0pt]
\[
  V = \pi \int_a^b f^2(x) - g^2(x) \text{ dx}
\]\\[0pt]
if the functions rotate around the \(y\) axis, the formula is\\[0pt]
\[
  V = 2\pi \int_a^b x\left(f^2(x) - g^2(x)\right) \text{ dx}
\]\\[0pt]
\begin{my_example}
we start with a simple example, consider the function \(y = x\)\\[0pt]

\includesvg{/Users/mahmooz/brain/out/dbBobk}\\[0pt]

\includesvg{/Users/mahmooz/brain/out/AeXpjL}\\[0pt]

to find the volume of this cone we use the disc integration formula, in this case \(g(x)=0\)\\[0pt]
assume the function stopped at \(x=a\) and started at \(x=0\) so we find the volume in the interval \([0,a]\)\\[0pt]
\begin{align*}
  V &= \pi \int_0^a f^2(x) - g^2(x) \text{ dx}\\
  &= \pi \int_0^a x^2 - 0^2 \text{ dx}\\
  &= \pi\left(\frac{x^3}{3}\Big|_0^a\right)\\
  &= \pi\left(\frac{a^3}{3}\right)\\
  &= \frac{a^3}{3} \cdot \pi
\end{align*}
\end{my_example}

\subsection{arc length}
\label{sec:org8075c71}
\begin{definition}
the length of the curve of a continuous function \(f(x)\) in the interval \([a,b]\) is defined as:\\[0pt]
\[
  L = \int_a^b \sqrt{1+(f'(x))^2} \text{ dx}
\]\\[0pt]
\begin{my_example}
find the length of the following curve in the interval \([0,3]\)\\[0pt]
\[
  f(x) = \frac13{(x^2+2)}^{\frac32}
\]\\[0pt]
first we find the derivative\\[0pt]
\[
  f'(x) = \frac13 \cdot \frac32{(x^2+2)}^{\frac12} \cdot 2x = x{(x^2+2)}^{\frac12}
\]\\[0pt]
we use the formula:\\[0pt]
\begin{align*}
  L &= \int_0^3 \sqrt{1+x^2(x^2+2)} \text{ dx}\\
  &= \int_0^3 \sqrt{1+x^4+2x^2} \text{ dx}\\
  &= \int_0^3 \sqrt{{(x^2+1)}^{2}} \text{ dx}\\
  &= \int_0^3 x^2+1 \text{ dx}\\
  &= \frac{x^3}{3}+x \Big|_0^3\\
  &= \frac{3^3}{3}+3\\
  &= 12
\end{align*}
\end{my_example}

\begin{my_example}
find the length of the curve of the function \(f(x) = \frac{x^4}{8} + \frac{1}{4x^2}\) in the interval \([1,3]\)\\[0pt]
\begin{align*}
  f(x) &= \frac{x^4}{8} + \frac{1}{4x^2}\\
  f(x) &= \frac{x^4}{8} + \frac{x^{-2}}{4}\\
  f'(x) &= \frac{4x^3}{8} + \frac{-2x^{-3}}{4}\\
  f'(x) &= \frac{x^3}{2} - \frac{x^{-3}}{2}\\
  {(f'(x))}^2 &= \left(\frac{x^3}{2} - \frac{x^{-3}}{2}\right)^2\\
  {(f'(x))}^2 &= \left(\frac{x^3}{2}\right)^2 - \frac{x^3 \cdot x^{-3}}{2} + \left(\frac{x^{-3}}{2}\right)^2\\
  {(f'(x))}^2 &= \left(\frac{x^3}{2}\right)^2 - \frac12 + \left(\frac{x^{-3}}{2}\right)^2\\
  L &= \int_a^b \sqrt{1+(f'(x))^2} \text{ dx}\\
  &= \int_a^b \sqrt{1+\left(\frac{x^3}{2}\right)^2 - \frac12 + \left(\frac{x^{-3}}{2}\right)^2} \text{ dx}\\
  &= \int_a^b \sqrt{\left(\frac{x^3}{2}\right)^2 + \frac12 + \left(\frac{x^{-3}}{2}\right)^2} \text{ dx}\\
  &= \int_a^b \sqrt{\left(\frac{x^3}{2}+\frac{x^{-3}}{2}\right)^2} \text{ dx}\\
  &= \int_a^b \frac{x^3}{2}+\frac{x^{-3}}{2} \text{ dx}\\
  &= \frac{x^4}{8}+\frac{x^{-2}}{-4} \Big|_a^b\\
  &= \frac{x^4}{8}+\frac{x^{-2}}{-4} \Big|_1^3\\
  &= \frac{3^4}{8}+\frac{3^{-2}}{-4} - \left(\frac{1^4}{8}+\frac{1^{-2}}{-4}\right)\\
  &= \frac{92}{9}
\end{align*}
\end{my_example}

\begin{my_example}
find the length of the curve of the function \(f(x) = \ln(\sin x)\) in the interval \(\left[\frac{\pi}{6},\frac{\pi}{2}\right]\)\\[0pt]
\begin{align*}
  f(x) &= \ln(\sin x)\\
  f'(x) &= \frac{1}{\sin x} \cdot \cos x = \cot x\\
  L &= \int_{\frac{\pi}{6}}^{\frac{\pi}{2}} \sqrt{1+(f'(x))^2} \text{ dx}\\
  &= \int_{\frac{\pi}{6}}^{\frac{\pi}{2}} \sqrt{1+\cot^2 x} \text{ dx}\\
  &= \int_{\frac{\pi}{6}}^{\frac{\pi}{2}} \sqrt{\frac{1}{\sin^2 x}} \text{ dx}\\
  &= \int_{\frac{\pi}{6}}^{\frac{\pi}{2}} \frac{1}{\sin x} \text{ dx}\\
  &= \int_{\frac{\pi}{6}}^{\frac{\pi}{2}} \frac{1}{\sin x} \text{ dx}\\
  &= \ln\tan\frac{x}{2} \Big|_{\frac{\pi}{6}}^{\frac{\pi}{2}}\\
  &= \ln\tan\frac{\pi}{4} - \ln\tan\frac{\pi}{12}\\
  &= 0 - \ln(2-\sqrt3)\\

\subsection{riemann sum}
\label{sec:org54fe05e}
\textbf{Riemann sum} can be applied to \hyperref[sec:org53be8a4]{integrable function}s to find area\\[0pt]
consider the area below the function \(e^{-x}\) in the interval \([1,2]\)\\[0pt]
we split the shape into rectangles that are as narrow as possible as the more rectangles we use the more accurate the result\\[0pt]
the following image shows the area split into \(n=8\) rectangles\\[0pt]
\includesvg{/Users/mahmooz/brain/out/JAIRcvb}\\[0pt]

you can probably tell that the sum of the areas of these rectangles isnt really close to the area we want to find to be more accurate we increase the number of rectangles to \(n=70\) and we get a nicer result:\\[0pt]

\includesvg{/Users/mahmooz/brain/out/5SZ0x8R}\\[0pt]

you can probably already tell that since this is a "narrower" coverage of the area, the sum of the rectangles would be closer to the value of the actual area\\[0pt]
and now that we know that the more rectangles we have the more accurate we get, we know we should be aiming at \(\infty\) rectangles\\[0pt]
the width of a rectangle would be \(\frac{1}{n}\) and the height would be \(f\left(\frac{k}{n}\right)\) where \(k\) is the index of the rectangle starting at 1, the sum of the area \(S\) of the rectangles would be:\\[0pt]
\[
  S = f\left(\frac{1}{n}\right) \cdot \frac{1}{n} + f\left(\frac{2}{n}\right) \cdot \frac{1}{n} + \cdots + f\left(\frac{n}{n}\right) \cdot \frac{1}{n}
\]\\[0pt]
we factor \(\frac{1}{n}\) out\\[0pt]
\[
  S = \frac{1}{n} \left(f\left(\frac{1}{n}\right) + f\left(\frac{2}{n}\right) + \cdots + f\left(\frac{n}{n}\right)\right)
\]\\[0pt]
and since we want \(n\) to be as big as possible, we take \(n \to \infty\)\\[0pt]
\[
  S = \lim_{n\to\infty} \frac{1}{n} \left(f\left(\frac{1}{n}\right) + f\left(\frac{2}{n}\right) + \cdots + f\left(\frac{n}{n}\right)\right)
\]\\[0pt]

\subsubsection{direct comparison test}
\label{sec:orgbfc2763}
\begin{definition}
given \(f(x) \geq g(x) \geq 0\) and it follows that \(\int_1^\infty g(x) \text{ dx} \leq \int_1^\infty f(x) \text{ dx}\), if \(\int_1^\infty f(x) \text{ dx}\) converges then \(\int_1^\infty g(x) \text{ dx}\) converges too and if \(\int_1^\infty g(x) \text{ dx}\) diverges then \(\int_1^\infty f(x) \text{ dx}\) diverges too\\[0pt]

\begin{my_example}
if we wanted to know if the integral \(\int_1^\infty \frac{1}{x^3+7} \text{ dx}\) diverges or converges we look at the expression we arrived at in the last lemma\\[0pt]
we take the function \(\frac{1}{x^3}\) which corresponds to the first case in that lemma where \(a > 1\) which means this function converges\\[0pt]
and since we know \(\frac{1}{x^3+7} < \frac{1}{x^3}\) and \(\int_1^\infty \frac{1}{x^3} \text{ dx}\) converges then we know \(\int_1^\infty \frac{1}{x^3+7} \text{ dx}\) converges too\\[0pt]
\end{my_example}

\begin{my_example}
check if the following integral diverges/converges\\[0pt]
\[
  \int_2^\infty \frac{1}{\ln{x}} \text{ dx}
\]\\[0pt]
we consider the function \(\frac1x\), we know \(\ln{x} \leq x\) is always true, if we were to turn them into fractions we would get \(\frac{1}{\ln{x}} \geq \frac{1}{x}\)\\[0pt]
and if we look at the same lemma from before we know \(\int_2^\infty \frac{1}{x} \text{ dx}\) diverges because it corresponds to the case where \(a=1\)\\[0pt]
and since \(\int_2^\infty \frac{1}{x} \text{ dx}\) diverges and \(\int_2^\infty \frac{1}{x} \text{ dx} \leq \int_2^\infty \frac{1}{\ln{x}} \text{ dx}\) then we know \(\int_2^\infty \frac{1}{\ln{x}} \text{ dx}\) diverges\\[0pt]
\end{my_example}

\begin{my_example}
consider the integral:\\[0pt]
\[
  \int_5^\infty \frac{1}{x\ln^2 x} \text{ dx}
\]\\[0pt]
we use \hyperref[sec:orgf2294be]{integration by substitution} where \(t = \ln x\)\\[0pt]
\(\text{ dt} = \frac1x \text{ dx} \Longrightarrow \text{ dx} = x\text{ dt}\)\\[0pt]
\begin{align*}
  \int \frac{1}{x\ln^2 x} \text{ dx} &= \int \frac{1}{xt^2} \cdot x \text{ dt}\\
  &= \int \frac{1}{t^2} \text{ dt}\\
  &= \int t^{-2} \text{ dt}\\
  &= -t^{-1}\\
  &= \frac{-1}{t}\\
  &= \frac{-1}{\ln x}\\
\end{align*}
we substitute the interval:\\[0pt]
\begin{align*}
  \int_5^\infty \frac{1}{x\ln^2 x} \text{ dx} &= \frac{-1}{\ln x} \Big|_5^\infty\\
  &= \underbrace{\frac{-1}{\ln \infty}}_{\clap{diverges to 0}} - \frac{-1}{\ln 5}\\
  &= \frac{1}{\ln 5}
\end{align*}
we found that the given integral converges to \(\frac{1}{\ln 5}\) (meaning the area between the given function and the \(x\) axis is equal to \(\frac{1}{\ln 5}\)).\\[0pt]
\end{my_example}

\begin{my_example}
this example shows another way to go about it\\[0pt]
consider the integral\\[0pt]

\(t = x^2 + 3 \Longrightarrow \text{dt} = 2x \text{ dx}\)\\[0pt]

\begin{align*}

  &= \frac{t^{11}}{11} \Bigg|_7^{52}\\
\end{align*}

\subsubsection{limit comparison test}
\label{sec:org572bf4d}
\begin{definition}
given \(f(x),g(x) \geq 0\), if:\\[0pt]
\[
  \lim_{x\to\infty} \frac{f(x)}{g(x)} = L,\ 0 < L < \infty
\]\\[0pt]
then the integrals of \(\int_1^\infty f(x) \text{ dx}\) and \(\int_1^\infty g(x) \text{ dx}\) either both diverge or both converge\\[0pt]

\begin{my_example}
consider the following integral\\[0pt]
\[
  \int_1^\infty \frac{2x^2 + 3}{5x^4 + x^2 - 9} \text{ dx}
\]\\[0pt]
to tell whether this integral is divergent/convergent we devide it by \(\frac{1}{x^2}\) because \(\frac{x^2}{x^4} = \frac{1}{x^2}\) is the biggest power in \(\frac{2x^2 + 3}{5x^4 + x^2 - 9}\)\\[0pt]
\[
  \lim_{x\to\infty} \frac{\frac{2x^2 + 3}{5x^4 + x^2 - 9}}{\frac{1}{x^2}} = \lim_{x\to\infty} \frac{2x^4+3x^2}{5x^4+x^2-9} = \frac25
\]\\[0pt]
and since we know \(\int_1^\infty \frac{1}{x^2} \text{ dx}\) is convergent we know \(\int_1^\infty \frac{2x^2 + 3}{5x^4 + x^2 - 9} \text{ dx}\) is convergent too\\[0pt]
\end{my_example}
\end{definition}
\subsubsection{Dirichlet's test}
\label{sec:org6f3d3ae}
\begin{definition}
given an \hyperref[sec:org53be8a4]{integrable function} \(f(x)\) for all \(x\) such that:\\[0pt]
\[
  \left|\int_a^b f(x) \text{ dx}\right| \leq M
\]\\[0pt]
where \(M\) is some constant, and given some monotonically decreasing function \(g(x)\) such that \(\lim_{x\to\infty} g(x) = 0\) then the integral \(\int_a^\infty f(x) \cdot g(x) \text{ dx}\) converges\\[0pt]
\begin{my_example}
consider the integral:\\[0pt]
\[
  \int_1^\infty \frac{\sin x}{x} \text{ dx}
\]\\[0pt]
we use this test to determine whether this integral diverges or converges\\[0pt]
\begin{gather*}
  \int_1^\infty \frac{\sin x}{x} \text{ dx} = \int_1^\infty \sin x \cdot \frac1x \text{ dx}\\
  \left|\int_a^b \sin x \text{ dx}\right| = \left|-\cos x\big|_a^b\right| = |-\cos b + \cos a| \leq 2
\end{gather*}
we take \(g(x) = \frac1x\), we know \(\lim_{x\to\infty} g(x) = 0\) and \(g'(x)=-\frac1{x^2} \leq 0\) therefore \(g(x)\) is a monotonically decreasing function therefore we can use it in our Dirichlet test and therefore \(\int_a^\infty f(x) \cdot g(x) \text{ dx}\) converges therefore the original integral \(\int_1^\infty \frac{\sin x}{x} \text{ dx}\) converges\\[0pt]
\end{my_example}

\begin{my_example}
\begin{align*}
  \int_1^\infty \frac{\cos^2 x}{x} \text{ dx} &= \int_1^\infty \frac{1+\cos 2x}{2x} \text{ dx}\\
  &= \frac12\int_1^\infty \frac{1}{x} \text{ dx} + \frac12\int_1^\infty \frac{\cos 2x}{x} \text{ dx}\\
\end{align*}
we take the second integral \(\int_1^\infty \frac{\cos 2x}{x} \text{ dx}\), this integral is of 2 functions:\\[0pt]
\begin{gather*}
  \left|\int_a^b \cos 2x \text{ dx}\right| = \left|\frac{\sin 2x}{2}\Bigg|_a^b\right| = \left|\frac{\sin 2b}{2} - \frac{\sin 2a}{2}\right| \leq 1\\
  g(x) = \frac1x,\ g'(x) = \frac{-1}{x^2} < 0,\ \lim_{x\to\infty} g(x) = 0
\end{gather*}
therefore the integral of the second function converges\\[0pt]
and the integral of the first function \(\frac1x\) diverges\\[0pt]
therefore the sum of those integrals converges\\[0pt]
\end{my_example}

\begin{my_example}
check whether the following integral converges\\[0pt]
\[
  \int_1^\infty x^4e^{-x} \cos x \text{ dx}
\]\\[0pt]
we take \(f(x) = \cos x\):\\[0pt]
\[
  \int_a^b \cos x \text{ dx} = \left|\sin x\big|_a^b\right| = \left|\sin b - \sin a\right| \leq 2
\]\\[0pt]
and we take \(g(x) = \frac{x^4}{e^x}\) and we know \(\lim_{x\to\infty} g(x) = 0\) since \(e^x\) outgrows any polynomial\\[0pt]
\[
  g'(x) = \frac{4x^3e^x-e^xx^4}{{(e^x)}^2} = \frac{x^3e^x(4-x)}{e^{2x}}
\]\\[0pt]
\(g'(x)\) is equal to 0 only in \(x=0,4\), we can substitute variables in both intervals \((0,4),(4,\infty]\) to find that in the interval \((0,4)\) the function is increasingly monotonic while in \((0,\infty]\) its decreasingly monotonic which is what we're looking for here\\[0pt]
therefore we split this integral in 2:\\[0pt]
\[
  \int_1^\infty x^4e^{-x} \cos x \text{ dx} = \underbrace{\int_1^4 x^4e^{-x}\cos x \text{ dx}}_{\text{finite}} + \underbrace{\int_4^\infty x^4e^{-x}\cos x \text{ dx}}_{\text{converges}}\\
\]\\[0pt]
the second part of the integral converges because it abides by Dirichlet's and the first part is just a finite definite integral that evaluates to some constant\\[0pt]
therefore this integral converges because the sum of 2 finite constants is a constant\\[0pt]
\[
 \int_1^\infty \frac{\sin^2 x}{x^3} \text{ dx} \leq \underbrace{\int_1^\infty \frac{1}{x^3} \text{ dx}}_{\text{converges}}
\]\\[0pt]
therefore the first integral also converges because its smaller\\[0pt]
\end{my_example}

\begin{question}
\[
  \int_1^\infty \sin(x^2) \text{ dx}
\]\\[0pt]
\begin{answer}
we use \hyperref[sec:orgf2294be]{integration by substitution}, let \(t = x^2 \Longrightarrow \text{dt} = 2x\text{ dx} \Longrightarrow \text{dx} = \frac{\text{dt}}{2x}\), and the interval in this case is the same for \(t\) and \(x\)\\[0pt]
\begin{align*}
  \int_1^\infty \sin(x^2) \text{ dx} &= \int_1^\infty \sin t \frac{\text{dt}}{2x}\\
  &= \frac12\int_1^\infty \frac{\sin t}{\sqrt{t}} \text{ dt}\\
\end{align*}
we can notice that the integral we arrived at can be solved using \hyperref[sec:org6f3d3ae]{Dirichlet's test}\\[0pt]
we pick \(f(t) = \sin t\):\\[0pt]
\[
  \left|\int_a^b \sin t \text{ dt}\right| = \left|-\cos t\big|_a^b\right| = |-\cos b + \cos a| \leq 2
\]\\[0pt]
we pick \(g(t) = \frac{1}{\sqrt{t}} = t^{-\frac12}\)\\[0pt]
\[
  g'(t) = -\frac12 t^{-\frac32} < 0
\]\\[0pt]
\(g(t)\) is a monotonically decreasing function and \(\lim_{t\to\infty} \frac{1}{\sqrt t} = 0\) therefore according to \hyperref[sec:org6f3d3ae]{Dirichlet's test} \(\frac12\int_1^\infty \frac{\sin t}{\sqrt{t}} \text{ dt}\) converges therefore \(\int_1^\infty \sin(x^2) \text{ dx}\) converges too\\[0pt]
\end{answer}
\end{question}

we take \(g(x) = \frac{\ln^3 x}{x}\), we use lhopital to find the limit\\[0pt]
\[
  \lim_{x\to\infty} \frac{\ln^3 x}{x} \Rightarrow \lim_{x\to\infty} \frac{3\ln^2 x \cdot \frac1x}{1} = \lim_{x\to\infty} \frac{3\ln^2 x}{x} \Rightarrow \lim_{x\to\infty} \frac{6\ln x \cdot \frac1x}{1} = \lim_{x\to\infty} \frac{6\ln x}{x} = \lim_{x\to\infty} \frac6x = 0
\]\\[0pt]
or we can say that \(x\) outgrows \(\ln^3 x\) therefore the function converges to 0, we find \(g'(x)\):\\[0pt]
\[
  g'(x) = \frac{3\ln^2 x \cdot \frac1x \cdot x - \ln^3 x}{x^2} = \frac{\ln^2 x(3-\ln x)}{x^2}
\]\\[0pt]
\(\frac{\ln^2 x(3-\ln x)}{x^2} = 0\) gives us the 2 solutions \(x=1,x=e^3\), and if we were to substitute \(x > e^3\) we would find that the function \(g(x)\) is monotonically decreasing in the interval \((e^3, \infty]\) and monotonically increasing in \((1,e^3)\), therefore these functions abide by \hyperref[sec:org6f3d3ae]{Dirichlet's test}'s requirements therefore we write:\\[0pt]
\[
  \int_8^\infty \frac{\ln^3 x \cdot \cos\frac{x}{3}}{x} \text{ dx} = \underbrace{\int_8^{e^3} \frac{\ln^3 x \cdot \cos\frac{x}{3}}{x} \text{ dx}}_{\text{finite}} + \underbrace{\int_{e^3}^{\infty} \frac{\ln^3 x \cdot \cos\frac{x}{3}}{x} \text{ dx}}_{\text{converges}}
\]\\[0pt]
finite + convergent = convergent therefore \(\int_8^\infty \frac{\ln^3 x \cdot \cos\frac{x}{3}}{x} \text{ dx}\) is convergent\\[0pt]
\end{answer}
\end{question}

\begin{question}
\[
  \int_8^{\infty} \frac{1}{x\ln x\ln\ln x} \text{ dx}
\]\\[0pt]
\begin{answer}
we use \hyperref[sec:orgf2294be]{substitution} where \(t = \ln\ln x \Rightarrow \text{dt} = \frac{1}{\ln x} \cdot \frac1x \Rightarrow \text{dx} = x\ln x \text{ dt}\)\\[0pt]
\(x = 8 \Rightarrow t = \ln\ln8 = 0.73,\ x = \infty \Rightarrow t = \ln\ln\infty = \infty\)\\[0pt]
\begin{align*}
  \int_8^{\infty} \frac{1}{x\ln x\ln\ln x} \text{ dx} &= \int_{0.73}^\infty \frac1{\cancel{x\ln x}t} \cancel{x\ln x} \text{ dt}\\
  &= \int_{0.73}^\infty \frac1t \text{ dt} &\text{diverges}
\end{align*}
therefore the original integral \(\int_8^{\infty} \frac{1}{x\ln x\ln\ln x} \text{ dx}\) diverges\\[0pt]
\end{answer}
\end{question}

\begin{definition}
consider the following series\\[0pt]
\begin{align}
  \sum_{n=1}^\infty a_n &= a_1 + a_2 + \cdots + a_n &\text{diverges, behaves like } f(x) = x\\
  \sum_{n=1}^\infty \frac1n &= \frac11 + \frac12 + \cdots + \frac1n &\text{diverges,  behaves like } f(x)=\frac1x\\
  \sum_{n=1}^\infty \frac1{n^2} &= \frac1{1^2} + \frac1{2^2} + \cdots + \frac1{n^2} &\text{converges,  behaves like } f(x)=\frac1{x^2}\\
  \sum_{n=1}^\infty \frac1{3^n} &= \frac1{3^1} + \frac1{3^2} + \cdots + \frac1{3^n} &\text{diverges, this is an infinite decreasing series}\\
  \sum_{n=1}^\infty 2n + 3 &= 5 + 7 + \cdots + 2n+3 &\text{diverges}\\
  \sum_{n=1}^\infty {(-1)}^n &= -1 + 1 + -1 + 1 + \cdots + {(-1)}^n &\text{diverges, it doesnt converge to a constant}
\end{align}
\begin{my_example}
consider the series:\\[0pt]
\[
  \sum_{n=1}^\infty \frac1{n(n+1)} = \sum_{n=1}^\infty \frac1n - \frac1{n+1} = \frac11 - \frac12 + \frac12 - \frac13 + \frac13 - \frac14 + \cdots + \frac1n - \frac1{n+1} = 1 - \frac{1}{n+1}

this series converges to 1 as it ends with \(1 - \frac{1}{n+1}\)\\[0pt]
\end{my_example}

\begin{my_example}

this series diverges\\[0pt]

\subsubsection{direct comparison test}
\label{sec:orgdc02fd7}
\begin{definition}
given the infinite series \(\sum^\infty b_n, \sum^\infty a_n\) such that \(|a_n| \leq |b_n|\) for sufficiently large \(n\) then:\\[0pt]
\begin{enumerate}
\item if the series \(b_n\) converges then the series \(a_n\) also converges\\[0pt]
\item if the series \(a_n\) diverges then the series \(b_n\) also diverges\\[0pt]
\end{enumerate}

\begin{my_example}
\begin{align*}
  \sum^\infty \frac{1}{n^5+17} \leq& \sum^\infty \frac{1}{n^5}\\
  \text{converges} \impliedby& \text{converges}
\end{align*}
\end{my_example}

\begin{my_example}
does this series converge or diverge?\\[0pt]
\[
  \sum_{n=2}^\infty \frac{1}{\ln n}
\]\\[0pt]
it diverges because \(\frac{1}{\ln x} > \frac{1}{n}\) and \(\sum \frac{1}{n}\) diverges\\[0pt]
\end{my_example}
\end{definition}
\subsubsection{limit comparison test}
\label{sec:org793987b}
\begin{definition}
if \(\{a_n\},\{b_n\} > 0\) (all elements of the two sequences are positive) and the limit \(\lim_{n\to\infty} \frac{a_n}{b_n}\) \hyperref[sec:org0440354]{exists} and is non-zero, then \(\sum_{n=1}^\infty a_n\) diverges if and only if \(\sum_{n=1}^\infty b_n\) diverges\\[0pt]
\begin{my_example}
\[
  \sum \frac{3n^3+7n^2-2}{7n^5+4n^2-3}
\]\\[0pt]
we divide by \(\frac{1}{n^2}\)\\[0pt]
\[
  \lim_{n\to\infty} \frac{\frac{3n^3+7n^2-2}{7n^5+4n^2-3}}{\frac{1}{n^2}} = \lim_{n\to\infty} \frac{3n^5+7n^4-2n^2}{7n^5+4n^2-3} = \frac37
\]\\[0pt]
we know \(\sum \frac{1}{n^2}\) converges therefore \(\sum \frac{3n^3+7n^2-2}{7n^5+4n^2-3}\) converges too\\[0pt]
\end{my_example}
\end{definition}
\subsubsection{Ratio test}
\label{sec:orgcc05a6d}
\begin{definition}
suppose there exists \(r\) such that\\[0pt]
\[
  \lim_{n\to\infty} \left|\frac{a_{n+1}}{a_n}\right| = r
\]\\[0pt]
If \(r < 1\), then the series \(\sum a_n\) is convergent. If \(r > 1\), then the series is divergent. If \(r = 1\), the \textbf{ratio test} is inconclusive, and the series may converge or diverge, and we may use the \hyperref[sec:org88b6001]{root test} which might succeed where the \hyperref[sec:orgcc05a6d]{ratio test} fails\\[0pt]

\begin{my_example}
consider \(\sum \frac1n\) which we already know is divergent, \(\lim_{n\to\infty} \frac{\frac{1}{n+1}}{\frac{1}{n}} = \lim_{n\to\infty} \frac{n}{n+1} = 1\) which means the test is inconclusive (fails)\\[0pt]
\end{my_example}

\begin{my_example}
consider \(\sum \frac{1}{3^n}\)\\[0pt]
\[
  \frac{a_{n+1}}{a_n} = \frac{3^n}{3^{n+1}} = \frac{3^n}{3 \cdot 3^n} = \frac13 < 1
\]\\[0pt]
which means \(\sum \frac{1}{3^n}\) converges\\[0pt]
\end{my_example}

\begin{my_example}
consider the series:\\[0pt]
\[
  \sum \frac{(n!)^2}{(2n)!}
\]\\[0pt]
we use the \hyperref[sec:orgcc05a6d]{ratio test}\\[0pt]
\begin{align*}
  \frac{a_{n+1}}{a_n} &= \frac{((n+1)!)^2}{(2(n+1))!} \cdot \frac{(2n)!}{(n!)^2}\\
  &= \frac{(n+1)^2 \cdot (n!)^2}{(2n+2)!} \cdot \frac{(2n)!}{(n!)^2}\\
  &= \frac{(n+1)^2 \cdot \cancel{(n!)^2}}{(2n+2)(2n+1)\cancel{((2n)!)}} \cdot \frac{\cancel{(2n)!}}{\cancel{(n!)^2}}\\
  &= \frac{n^2+2n+1}{4n^2+6n+2}\\
  &\implies \frac14
\end{align*}
which means this series is convergent\\[0pt]
\end{my_example}

\begin{my_example}
consider the series\\[0pt]
\[
  \sum \frac{2^n \cdot n!}{n^n}
\]\\[0pt]
we make use of the \hyperref[sec:orgcc05a6d]{ratio test}\\[0pt]
\begin{align*}
  \frac{a_{n+1}}{a_n} &= \frac{2^{n+1} \cdot (n+1)!}{(n+1)^{n+1}} \cdot \frac{n^n}{2^n \cdot n!}\\
  &= \frac{2 \cdot \cancel{2^n \cdot (n+1)n!} \cdot n^n}{\cancel{(n+1)}(n+1)^n \cdot \cancel{2^n \cdot n!}}\\
  &= \frac{2n^n}{(n+1)^n}\\
  &= 2\left(\frac{n}{n+1}\right)^n
\end{align*}
we find the limit of \(\left(\frac{n}{n+1}\right)^n\) using euler\\[0pt]
\begin{gather*}
  \lim_{n\to\infty} \left(\frac{n}{n+1}\right)^n = e^L\\
  L = \lim_{n\to\infty} \left(\frac{n}{n+1} - 1\right)n = \lim_{n\to\infty} \left(\frac{-1}{n+1}\right)n = \lim_{n\to\infty} \frac{-n}{n+1} = -1
\end{gather*}
we continue where we left off:\\[0pt]
\[
 \lim_{n\to\infty} 2\left(\frac{n}{n+1}\right)^n = 2e^{-1} = \frac{2}{e} < 1
\]\\[0pt]
which means this series is convergent\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \sum_{n=1}^\infty \sin\frac1n
\]\\[0pt]
we know \(\sin x\) behaves like \(x\) so we divide by \(\frac1n\) and use the \hyperref[sec:org793987b]{limit comparison test}\\[0pt]
\[
  \lim_{n\to\infty} \frac{\sin\frac1n}{\frac1n} = 1
\]\\[0pt]
which means the series is divergent because \(\sum \frac1n\) diverges\\[0pt]
note that \(\sum_{n=1}^\infty \sin^2\frac1n\) is convergent because we would divide by \(\frac1{n^2}\) whose series converges\\[0pt]
\end{my_example}
\end{definition}
\subsubsection{Root test}
\label{sec:org88b6001}
\begin{definition}
let\\[0pt]
\[
  r = \lim_{n\to\infty} \sqrt[n]{|a_n|}
\]\\[0pt]
if \(r < 1\), the series converges, if \(r > 1\), then the series diverges, if \(r = 1\), the root test is inconclusive, and the series may converge or diverge.\\[0pt]

\begin{note}
this is a consequence of the following equality:\\[0pt]
\[
  \lim \frac{a_{n+1}}{a_n} = \lim \sqrt[n]{a_n}
\]\\[0pt]
\end{note}

\begin{my_example}
consider\\[0pt]
\[
  \sum \frac{1}{3^n}
\]\\[0pt]
we use the \hyperref[sec:org88b6001]{root test}\\[0pt]
\[
\sqrt[n]{a_n} = \sqrt[n]{\frac{1}{3^n}} = \frac13 < 1
\]\\[0pt]
which means that the series converges\\[0pt]
\end{my_example}

\begin{my_example}
consider\\[0pt]
\[
\sum \frac{1}{\left(\frac{n+1}{n+3}\right)^{n(n+1)}}
\]\\[0pt]
we apply the \hyperref[sec:org88b6001]{root test}\\[0pt]
\[
  \sqrt[n]{a_n} = \sqrt[n]{\frac{1}{\left(\frac{n+1}{n+3}\right)^{n(n+1)}}} = \frac{1}{\left(\frac{n+1}{n+3}\right)^{n+1}}
\]\\[0pt]
we find the limit of the fraction:\\[0pt]
\begin{gather*}
  \lim_{n\to\infty} \left(\frac{n+1}{n+3}\right)^{n+1} = e^L\\
  L = \lim_{n\to\infty} \left(\frac{n+1}{n+3}-1\right)(n+1) = \lim_{n\to\infty} \left(\frac{-2}{n+3}\right)(n+1) = \lim_{n\to\infty} \frac{-2n-2}{n+3} = -2
\end{gather*}
we go back to the original limit:\\[0pt]
\begin{gather*}
  \lim_{n\to\infty} \frac{1}{\left(\frac{n+1}{n+3}\right)^{n+1}} = \lim_{n\to\infty} \frac{1}{e^{-2}} = e^2 > 1
\end{gather*}
which means that the series diverges\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \sum \left(\frac{n}{n+1}\right)^{2n}
\]\\[0pt]
we apply the \hyperref[sec:org88b6001]{root test}\\[0pt]
\begin{gather*}
  \lim_{n\to\infty} \left(\frac{n}{n+1}\right)^{2n} = \frac{1}{e^L}\\
  a_n = \left(\frac{n}{n+1}\right)^{2n} \implies \frac{1}{e^L} \neq 0\\
  L \implies \left(\frac{n}{n+1}-1\right)2n = \left(\frac{-1}{n+1}\right)2n = \frac{-2n}{n+1} \implies -2\\
  \lim_{n\to\infty} \left(\frac{n}{n+1}\right)^{2n} = \frac{1}{e^{-2}} = e^2 > 1
\end{gather*}
which means that the series diverges\\[0pt]
\end{my_example}
\end{definition}
\subsubsection{Integral test}
\label{sec:org8492316}
\begin{definition}
let \(f(x) > 0\) then the integral \(\int_1^\infty f(x) \text{ dx}\) and the series \(\sum_{n=1}^\infty f(n)\) converge or diverge together\\[0pt]

\begin{my_example}
\[
  \sum_{n=5}^\infty \frac{1}{n\ln n}
\]\\[0pt]
we use the \hyperref[sec:org8492316]{integral test} and compare the series to the integral \(\int_5^\infty \frac{1}{x\ln x} \text{ dx}\)\\[0pt]
\begin{gather*}
  t = \ln x\\
  \text{dt} = \frac1x \text{ dx}\\
  \text{dx} = x \text{ dt}\\
  x=5 \implies t = \ln5\\
  x=\infty \implies t=\ln\infty=\infty\\
  \int_{\ln5}^\infty \frac{1}{xt}x \text{ dt} = \underbrace{\int_{\ln5}^\infty \frac{1}{t} \text{ dt}}_{\text{diverges}}
\end{gather*}
which means that the series \(\sum_{n=5}^\infty \frac{1}{n\ln n}\) diverges too\\[0pt]
\end{my_example}

\begin{my_example}
using both \hyperref[sec:orgcc05a6d]{ratio test} and \hyperref[sec:orgdc02fd7]{direct comparison test}\\[0pt]
\begin{gather*}
  \sum \frac{1!+2!+3!+\cdots+n!}{(2n)!} \leq \sum \frac{n!+n!+n!+\cdots+n!}{(2n)!} = \sum \frac{n\cdot n!}{(2n)!}\\
  \frac{a_{n+1}}{a_n} = \frac{(n+1)(n+1)!}{(2(n+1))!} \cdot \frac{(2n)!}{n\cdot n!} = \frac{(n+1)\cancel{(n+1)n! \cdot (2n)!}}{\cancelto{2}{(2n+2)}(2n+1)\cancel{(2n)!} \cdot n \cdot \cancel{n!}} = \frac{n+1}{2(2n+1)n} = \frac{n+1}{4n^2+2n} \implies 0 < 1
\end{gather*}
which means the series converges\\[0pt]
\end{my_example}

\begin{my_example}
\begin{gather*}
  \sum_{n=1}^\infty 1-\cos\frac1n = \sum_{n=1}^\infty 2\sin^2\frac{1}{2n}\\
  \lim \frac{2\sin^2\frac{1}{2n}}{\frac{1}{n^2}} \implies 8
\end{gather*}
unfinished\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \sum_{n=7}^\infty = \frac{\ln^{-3} n}{n}
\]\\[0pt]
we apply the \hyperref[sec:org8492316]{integral test}\\[0pt]
\begin{gather*}
  t = \ln x\\
  \text{dt} = \frac1x \text{ dx}\\
  \text{dx} = x\text{ dt}\\
  x=7 \implies t=\ln7\\
  x=\infty \implies t=\ln\infty=\infty\\
  \int_{\ln7}^\infty \frac{1}{x\ln^3 x} \text{ dx} = \int_{\ln7}^\infty \frac{1}{xt^3} x \text{ dt} = \underbrace{\int_{\ln7}^\infty \frac{1}{t^3} \text{ dt}}_{\text{converges}}
\end{gather*}
which means the series converges too\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \sum \frac{(n+1)! + (n+2)! + \cdots + (2n)!}{(2n+3)!} \leq \sum \frac{n(2n)!}{(2n+3)(2n+2)(2n+1)(2n)!} \leq \sum \frac{n}{n \cdot n \cdot n} = \sum \frac{1}{n^2}
\]\\[0pt]
and since \(\sum \frac{1}{n^2}\) converges we know the original series converges because its smaller\\[0pt]
\end{my_example}

\begin{my_example}
\[
  \sum \frac{2^n}{\left(\frac{3n+2}{3n+1}\right)^{n(2n-3)}}
\]\\[0pt]
we use the \hyperref[sec:org88b6001]{Root test}:\\[0pt]
\[
  \sqrt[n]{\frac{2^n}{\left(\frac{3n+2}{3n+1}\right)^{n(2n-3)}}} = \frac{2}{\left(\frac{3n+2}{3n+1}\right)^{2n-3}} \implies \frac{2}{e^{\frac23}} > 1
\]\\[0pt]
which means the series diverges\\[0pt]
\end{my_example}
\end{definition}
\subsubsection{Raabe–Duhamel's test}
\label{sec:orgd8424da}
\begin{definition}
given \(\lim_{n\to\infty} \left(\frac{a_n}{a_{n+1}}-1\right)n = L\)\\[0pt]
\begin{enumerate}
\item if \(L < 1\) then \(\sum a_n\) diverges\\[0pt]
\item if \(L > 1\) then \(\sum a_n\) converges\\[0pt]
\item the test is inconclusive\\[0pt]
\end{enumerate}
we use this test when the \hyperref[sec:orgcc05a6d]{Ratio test} fails\\[0pt]

\begin{my_example}
\[
  \sum \frac{n!}{(5+1)(5+2)(5+3) \cdots (5+n)}
\]\\[0pt]
we try \hyperref[sec:orgcc05a6d]{Ratio test}:\\[0pt]
\begin{align*}
  \frac{a_{n+1}}{a_n} &= \frac{(n+1)!}{(5+1)(5+2) \cdots (5+n+1)} \cdot \frac{(5+1)(5+2) \cdots (5+n)}{n!}\\
  &= \frac{(n+1)n!}{(5+1)(5+2) \cdots (5+n+1)} \cdot \frac{(5+1)(5+2) \cdots (5+n)}{n!}\\
  &= \frac{n+1}{n+6}\\
  &\implies 1
\end{align*}
\hyperref[sec:orgcc05a6d]{Ratio test} failed because we got 1 so we apply \textbf{Raabe's test}:\\[0pt]
\begin{align*}
  \lim_{n\to\infty} \left(\frac{a_n}{a_{n+1}}-1\right)n &= \lim_{n\to\infty} \left(\frac{n+6}{n+1}-1\right)n\\
  &= \lim_{n\to\infty} \left(\frac{5}{n+1}\right)n\\
  &= \lim_{n\to\infty} \frac{5n}{n+1}\\
  &= 5 > 1
\end{align*}
we got \(5 > 1\) which means the series converges\\[0pt]
\end{my_example}
\end{definition}
\subsubsection{Alternating series test}
\label{sec:orgcbff953}
suppose the following statements are true:\\[0pt]

\begin{enumerate}
\item \(a_{n}\) are all positive,\\[0pt]
\item \(\lim_{n\to\infty} a_n = 0\)\\[0pt]
\item for every \(n\), \(a_{n+1} \leq a_n\)\\[0pt]
\end{enumerate}

then \(\sum_{n=k}^{\infty} (-1)^n a_n\) and \(\sum_{n=k}^{\infty} (-1)^{n+1} a_n\) are convergent series\\[0pt]

\begin{my_example}
\[
  \sum_{n=1}^{\infty} \frac{(-1)^n}{n} = \sum_{n=1}^{\infty} (-1)^n\frac{1}{n}
\]\\[0pt]
we check the conditions of the \hyperref[sec:orgcbff953]{Alternating series test}\\[0pt]
\begin{enumerate}
\item \(\frac{1}{n} \geq 0\)\\[0pt]
\item \(\lim_{n\to \infty} \frac{1}{n} = 0\)\\[0pt]
\item \(\frac{1}{n+1} < \frac{1}{n}\)\\[0pt]
\end{enumerate}
and since \(\frac{1}{n}\) meets all the conditions, we can apply the test to \(\sum_{n=1}^{\infty} (-1)^n\frac{1}{n}\) and find that it converges\\[0pt]
\end{my_example}
