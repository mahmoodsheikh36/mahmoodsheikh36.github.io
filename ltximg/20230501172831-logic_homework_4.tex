\begin{question}
given the functions\\
\begin{gather*}
  h(x,y,z) = \lnot x \land (\lnot y \to z)\\
  g(x,y,z) = x \to (y \to z)
\end{gather*}
\begin{subquestion}
is \(\{h\}\) a functionally complete set of connectives?\\
\begin{answer}
\(h(x,y,z) = \lnot x \land (\lnot y \to z) = \lnot x \land (y \lor z)\)\\
i think we can replicate F\\
\(h(x,x,x)=\lnot x \land (x \lor x) = \lnot x \land x = F\)\\
we cant replicate \href{20230422184813-negation.org}{negation} and \href{20230422184822-conjunction.org}{conjunction} (or \href{20230422184826-disjunction.org}{disjunction})\\
\end{answer}
\end{subquestion}
\begin{subquestion}
is \(\{g\}\) a functionally complete set of connectives?\\
\begin{answer}
\(g(x,y,z) = x \to (y \to z) = \lnot x \lor (\lnot y \lor z) = \lnot x \lor \lnot y \lor z\\\)\\
no\\
\end{answer}
\end{subquestion}
\begin{subquestion}
is \(\{h,g\}\) a functionally complete set of connectives?\\
\begin{answer}
probably, lets try\\
\begin{gather*}
  g(x,y,z) = x \to (y \to z) = \lnot x \lor (\lnot y \lor z) = \lnot x \lor \lnot y \lor z\\
  g(x,x,F=h(x,x,x)) = g(x,x,F) = \lnot x \lor \lnot x \lor F = \lnot x\\
  g(\lnot x,\lnot y,F) = x \lor y
\end{gather*}
it covers \(\{\lnot, \lor\}\) so it must be functionally complete\\
\end{answer}
\end{subquestion}

\begin{question}
given the functions\\
\begin{gather*}
  h(x,y,z) = x \to (y \to \lnot z)\\
  g(x,y,z) = (x \to y) \to \lnot z
\end{gather*}
\begin{subquestion}
is \(\{h\}\) a functionally complete set of connectives?\\
\begin{answer}
\(h(x,y,z) = x \to (y \to \lnot z) = \lnot x \lor (\lnot y \lor \lnot z)\)\\
we can implement negation:\\
\(h(x,x,x) = \lnot x\)\\
we can also implement disjunction:\\
\(h(\lnot x, \lnot y, \lnot x) = x \lor y\)\\
we implemented \(\{\lnot, \lor\}\) so its functionally complete\\
\end{answer}
\end{subquestion}
\begin{subquestion}
is \(\{g\}\) a functionally complete set of connectives?\\
\begin{answer}
\(h(x,y,z) = (x \to y) \to \lnot z = \lnot (\lnot x \lor y) \lor \lnot z = x \land y \lor \lnot z\)\\
no, cant implement anything\\
\end{answer}
\end{subquestion}
\begin{subquestion}
is \(\{h,g\}\) a functionally complete set of connectives?\\
\begin{answer}
yes, because \(\{h\}\) is functionally complete\\
\end{answer}
\end{subquestion}

\begin{question}
given the functions\\
\begin{gather*}
  h(x,y) = (\lnot x \land y) \lor (x \land \lnot y)\\
  g(x,y,z) = (\lnot x \to y) \to z
\end{gather*}

\begin{answer}

is \(\{g\}\) a functionally complete set of connectives?\\

\end{subquestion}
\begin{subquestion}
is \(\{h,g\}\) a functionally complete set of connectives?\\
\begin{answer}
maybe, lets try\\
\(g(x,F,F) = \lnot x \land T \lor F = \lnot x\)\\
we implemented negation\\
\(g(\lnot x, \lnot x, z) = x \lor z\)\\
we implemented \(\{\lnot, \lor\}\) so \(\{h,g\}\) is functionally complete\\
\end{answer}
\end{subquestion}

\begin{question}
given the functions\\
\begin{gather*}
  h(x,y,z) = ((x \to y) \to (y \to z))\\
  g(x,y,z) = \lnot ((x \to y) \to (x \to z))
\end{gather*}
\begin{subquestion}
is \(\{h\}\) a functionally complete set of connectives?\\
\begin{answer}
\(h(x,y,z) = (\lnot (\lnot x \lor y) \lor (\lnot y \lor z)) = x \land \lnot y \lor \lnot y \lor z = \lnot y \lor z\)\\
we can implement T, but no connectives\\
\(h(x,y,y) = \lnot y \lor y = T\)\\
\end{answer}
\end{subquestion}
\begin{subquestion}
is \(\{g\}\) a functionally complete set of connectives?\\
\begin{answer}
\(g(x,y,z) = \lnot (\lnot (\lnot x \lor y) \lor (\lnot x \lor z)) = ((\lnot x \lor y) \land \lnot (\lnot x \lor z)) = ((\lnot x \lor y) \land (x \land \lnot z))\)\\
we can implement F, but no connectives\\
\(g(x,x,x)=F\)\\
\end{answer}
\end{subquestion}
\begin{subquestion}
is \(\{h,g\}\) a functionally complete set of connectives?\\
\begin{answer}
we implement negation:\\
\(h(x,y,F)=\lnot y\)\\
then implement disjunction:\\
\(h(x,\lnot y,z)=y \lor z\)\\
so \(\{h,g\}\) is functionally complete\\
\end{answer}
\end{subquestion}
\end{question}
