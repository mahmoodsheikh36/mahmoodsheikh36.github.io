\begin{definition}
a \textbf{statement} can be true or false\\
\end{definition}
\begin{characteristic}
to prove that a property exists for every statement, we prove that\\
\begin{enumerate}
\item the property exists for every \href{20230401175954-boolean_variable.org}{boolean variable}\\
\item if the property exists for statement \(A\) then it also exists for \(\lnot A\)\\
\item if the property exists for the statements \(A\) and \(B\) then it also exists for \((A \land B),(A \lor B),(A \implies B),(A \iff B)\)\\
\end{enumerate}
\end{characteristic}
\begin{characteristic}
to represent a statement using a \href{data_structures.org}{binary tree}\\
\begin{itemize}
\item if the statement \(A\) is a boolean variable, draw a single node for \(A\)\\
\end{itemize}
\end{characteristic}
