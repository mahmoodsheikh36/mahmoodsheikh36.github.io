\begin{question}
10\% are rich, 5\% are famous, 2\% are rich and famous, what is the probability that of 10 people atleast one is famous or rich\\
\begin{answer}
A - rich\\
B - famous\\
\begin{align*}
  P(A \cup B) &= P(A) + P(B) - P(A \cap B)\\
  &= 0.1 + 0.05 - 0.02\\
  &= 0.13\\
  P\left(\overline{A} \cap \overline{B}\right) &= 1 - P(A \cup B)\\
  &= 1 - 0.13 = 0.87\\
  \text{answer} &= 1 - 0.87^{10}
\end{align*}
\end{answer}

\begin{question}
in a class theres 7 students, 4 girls and 3 boys, we pick 2 students, we denote by \(X\) the number of boys picked, what is the \href{20230121184721-expected_value.org}{expected value} and \href{20230113164748-variance.org}{variance} of \(X\)?\\
\begin{answer}
we use the \href{20230113165622-hypergeometric_distribution.org}{hypergeometric distribution}\\
\[
X = \begin{cases}
      0 & \frac{\binom{3}{0}\binom{7-3}{2-0}}{\binom{7}{2}} = \frac{2}{7}\\
      1 & \frac{\binom{3}{1}\binom{7-3}{2-1}}{\binom{7}{2}} = \frac{4}{7}\\
      2 & \frac{\binom{3}{2}\binom{7-3}{2-2}}{\binom{7}{2}} = \frac{1}{7}
    \end{cases}
\]\\
the expected value is \(0 \cdot \frac{2}{7} + 1 \cdot \frac{4}{7} + 2 \cdot \frac{1}{7} = \frac{6}{7}\)\\
the variance is\\
\[
\Var(X)=E\left(\left(X-\frac{6}{7}\right)^2\right)=\left(0-\frac{6}{7}\right)^2 \cdot \frac{2}{7}+\left(1-\frac{6}{7}\right)^2\cdot\frac{4}{7}+\left(2-\frac{6}{7}\right)^2\cdot\frac{1}{7}
\]\\
\end{answer}
\end{question}
\begin{question}
the number of people that arrive at a bus station in fifteen minutes is a random variable with a \href{20230121204918-poisson_distribution.org}{poisson distribution} with parameter \(\lambda=10\)\\
\begin{subquestion}
what is the probability that from 10:00 to 10:15 7 people arrive at the bus station?\\
\begin{answer}
\[
P(X=7)=\frac{10^7}{7!}e^{-10}
\]\\
\end{answer}
\end{subquestion}
\begin{subquestion}
what is the probability that from 20:15 to 20:30 at most 3 people arrive at the bus station?\\
\begin{answer}
\(\frac{10^0}{0!}e^{-10}+\frac{10^1}{1!}e^{-10}+\frac{10^2}{2!}e^{-10}+\frac{10^3}{3!}e^{-10}\)\\
\end{answer}
\end{subquestion}
\end{question}
