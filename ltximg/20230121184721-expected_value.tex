consider a \href{20230121192455-discrete_random_variable.org}{discrete random variable} \(X\) that receives the values \(X_1,X_2,\dots\) with the probabilities \(P_1,P_2,\dots\) respectively, the \textbf{expected value} of \(X\), denoted by \(\mathrm{E}(X)\), is defined as \(\mathrm{E}(X)=\sum_{i} X_iP_i\) such that the series is absolutely \href{calculus2.org}{convergent}\\[0pt]
\begin{characteristic}
if \(X\) is a constant random variable, i.e. it receives only one value \(C\) with a probability of 1 then \(\mathrm{E}(X)=C\)\\[0pt]
\end{characteristic}
\begin{characteristic}
if \(a\) is constant, then for each random variable whose expected value is finite \(\mathrm{E}(aX)=a\cdot \mathrm{E}(X)\)\\[0pt]
\end{characteristic}
\begin{characteristic}
let \(X\) and \(Y\) be two random variables defined over the same \href{probability.org}{probability space}, then\\[0pt]
\[
\mathrm{E}(X+Y)=\mathrm{E}(X)+\mathrm{E}(Y)
\]\\[0pt]
\begin{my_example}
let \(X \sim B(n,p)\), we look at \(X\) as the number of successes from a sequence of \(n\) independent trials\\[0pt]
\end{my_example}
\label{org8c37d7e}
\end{characteristic}
