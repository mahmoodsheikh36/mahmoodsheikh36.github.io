Simple DirectMedia Layer is a cross-platform development library designed to provide low level access to audio, keyboard, mouse, joystick, and graphics hardware via OpenGL and Direct3D\\
SDL officially supports Windows, Mac OS X, Linux, iOS, and Android. support for other platforms may be found in the source code\\
SDL is written in C, works natively with \href{cpp.org}{C++}, and there are bindings available for several other languages, including C\# and \href{20220809215936-python.org}{python} \cite{sdl}\\

\section{create window}
\label{sec:org18efafd}
\lstset{language=C++,label= ,caption= ,captionpos=b,numbers=none}
\begin{lstlisting}
#include <stdio.h>

#ifdef _WIN32
#include <SDL/SDL.h> /* Windows-specific SDL2 library */
#else
#include <SDL2/SDL.h> /* macOS and Linux-specific */
#endif

int main(int argc, char* argv[]) {
  SDL_Window *window;
  SDL_Init(SDL_INIT_VIDEO);

  window = SDL_CreateWindow(
    "sdl2 demo",                // window title
    SDL_WINDOWPOS_UNDEFINED,    // initial x position
    SDL_WINDOWPOS_UNDEFINED,    // initial y position
    640,                        // width, in pixels
    480,                        // height, in pixels
    0                           // flags
  );

  if (window == NULL) {
    printf("Could not create window: %s\n", SDL_GetError());
    return 1;
  }

  // the window is open, start handling events
  SDL_Event e;
  bool quit = false;
  while (!quit) {
    while (SDL_PollEvent(&e)) {
      if (e.type == SDL_QUIT) {
        quit = true;
      }
      if (e.type == SDL_KEYDOWN) {
        quit = true;
      }
      if (e.type == SDL_MOUSEBUTTONDOWN) {
        quit = true;
      }
    }
  }

  // pause execution for 3000 milliseconds
  // SDL_Delay(3000);

  // close and destroy the window
  SDL_DestroyWindow(window);

  // clean up
  SDL_Quit();
  return 0;
}
\end{lstlisting}

this code can be compiled with \texttt{g++ -lsdl2 myfile.cpp}\\

\section{basic rendering}
\label{sec:org46e99ed}
\url{https://wiki.libsdl.org/SDL\_Renderer}\\
\lstset{language=C++,label= ,caption= ,captionpos=b,numbers=none}
\begin{lstlisting}
#include <stdio.h>

#ifdef _WIN32
#include <SDL/SDL.h> /* Windows-specific SDL2 library */
#else
#include <SDL2/SDL.h> /* macOS and Linux-specific */
#endif

int main(int argc, char* argv[]) {
  SDL_Window *window;
  SDL_Renderer *renderer;
  SDL_Init(SDL_INIT_VIDEO);

  window = SDL_CreateWindow(
    "basic rendering",          // window title
    SDL_WINDOWPOS_UNDEFINED,    // initial x position
    SDL_WINDOWPOS_UNDEFINED,    // initial y position
    640,                        // width, in pixels
    480,                        // height, in pixels
    0                           // flags
  );
  if (window == NULL) {
    printf("Could not create window: %s\n", SDL_GetError());
    return 1;
  }

  renderer = SDL_CreateRenderer(window, -1, SDL_RENDERER_ACCELERATED);
  if (renderer == NULL) {
    printf("Could not create renderer: %s\n", SDL_GetError());
    return 1;
  }

  // some basic rendering
  SDL_Rect rect = {.x=30,.y=30,.w=50,.h=50};

  SDL_SetRenderDrawColor(renderer, 255, 0, 0, 255);
  SDL_RenderFillRect(renderer, &rect);

  SDL_SetRenderDrawColor(renderer, 255, 255, 255, 255);
  SDL_RenderDrawRect(renderer, &rect);

  SDL_RenderDrawLine(renderer, 0, 0, 30, 250);

  SDL_RenderPresent(renderer);

  // start handling events
  SDL_Event e;
  bool quit = false;
  while (!quit) {
    while (SDL_PollEvent(&e)) {
      if (e.type == SDL_QUIT) {
        quit = true;
      }
      if (e.type == SDL_KEYDOWN) {
        quit = true;
      }
      if (e.type == SDL_MOUSEBUTTONDOWN) {
        quit = true;
      }
    }
  }

  // pause execution for 3000 milliseconds
  // SDL_Delay(3000);

  // close and destroy the window
  SDL_DestroyWindow(window);

  // clean up
  SDL_Quit();
  return 0;
}
\end{lstlisting}
