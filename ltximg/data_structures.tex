\section{data structure}
\label{sec:org8f50ccd}
\begin{definition}
a \textbf{data structure} is a collection of data and operations on said data, the data is represented as objects, and each object \texttt{x} has:\\[0pt]
\begin{itemize}
\item \texttt{key(x)}: a key that is used to identify the object (usually a number that uniquely identifies the object)\\[0pt]
\item \texttt{info(x)} or \texttt{value(x)}: contains the data needed to store for the object (usually without the key)\\[0pt]
\end{itemize}
\begin{my_example}
for example, if \texttt{x} is a person, \texttt{key(x)} could be the person's ID, \texttt{value(x)} would be information on said person, like age, height, weight etc..\\[0pt]
\end{my_example}
\end{definition}

template <typename K, typename V>
class KeyValue {
private:
  K key;
  V value;

public:
  KeyValue(K key, V value) {
    this->key = key;
    this->value = value;
  }
  K get_key() {
    return this->key;
  }
  V get_value() {
    return this->value;
  }
  void set_value(V value) {
    this->value = value;
  }
  void set_key(K key) {
    this->key = key;
  }
  friend bool operator <(const KeyValue<K, V>& c1, const KeyValue<K, V>& c2) {
    return c1.key < c2.key;
  }
  friend bool operator >(const KeyValue<K, V>& c1, const KeyValue<K, V>& c2) {
    return c1.key > c2.key;
  }
};
\end{lstlisting}
\subsection{abstract data type}
\label{sec:orga3b31cf}
\begin{definition}
\textbf{abstract data types}, commonly abbreviated *ADT*s, are a way of classifying \hyperref[sec:org8f50ccd]{data structure}s based on how they are used and the behaviors they provide. they do not specify how the data structure must be implemented or laid out in memory, but simply provide a minimal expected interface and set of behaviors. for example, a \hyperref[sec:org1fa0f82]{stack} is an abstract data type that specifies a linear data structure with \hyperref[sec:org131c393]{LIFO} behavior. stacks are commonly implemented using \href{20220728190531-array.org}{array}s or \hyperref[sec:org3b67cf5]{linked list}s, but a needlessly complicated implementation using a \hyperref[sec:orge1ff544]{binary search tree} is still a valid implementation. to be clear, it is incorrect to say that stacks are arrays or vice versa. an array can be used as a stack. likewise, a stack can be implemented using an array.\\[0pt]
\begin{characteristic}
since abstract data types dont specify an implementation, this means its also incorrect to talk about the \href{20221130014441-time_complexity.org}{time complexity} of a given abstract data type. an associative array may or may not have \(O(1)\) average search times. An associative array that is implemented by a \hyperref[sec:org4f64d21]{hash table} does have \(O(1)\) average search times.\\[0pt]
\end{characteristic}

\begin{characteristic}
to further complicate matters, since certain abstract data types are almost always implemented with a particular data structure, some programmers will use the two terms interchangeably: for example, priority queue and heap, or associative array and hash table. the context in which the term is being used can usually provide distinction.\\[0pt]
\end{characteristic}
\end{definition}
\subsection{concrete data type}
\label{sec:org39a1299}
\begin{definition}
a \textbf{concrete data type} is the "concrete" and actual implementation of a given \hyperref[sec:orga3b31cf]{abstract data type}\\[0pt]
i.e. it is the actual \hyperref[sec:org8f50ccd]{data structure} we use in programming\\[0pt]
\end{definition}
\subsection{list}
\label{sec:org250337b}
\begin{definition}
a \textbf{list} is an \hyperref[sec:orga3b31cf]{abstract data type} that is dynamically sized (often automatically resized) \textbf{list of objects} that stores as many objects as necessary, each type of list has its own method of dynamic storage\\[0pt]
\end{definition}
\subsubsection{linked list}
\label{sec:org3b67cf5}
\begin{definition}
a \textbf{linked list} is a \hyperref[sec:org39a1299]{concrete data type} that is an implementation of a \hyperref[sec:org250337b]{list}\\[0pt]
\end{definition}
\paragraph{characteristics}
\label{sec:org55250f7}
\begin{itemize}
\item is the list \textbf{sorted}? the nodes are sorted by their \texttt{key} (usually in an ascending order)\\[0pt]
\item is the list \textbf{doubly/singly linked}? in a doubly linked list each node points to the next and previous node whereas in a singly linked list a node only points to the next\\[0pt]
\item does the list have a \textbf{tail}? a pointer \texttt{tail(x)} points to the last node in a linked list\\[0pt]
\item is the list \textbf{circular}? in a circular linked list the last node points to the first whereas in a non-circular linked list it points to NULL (nothing)\\[0pt]
\end{itemize}
\paragraph{functions}
\label{sec:org756421d}
\begin{itemize}
\item \texttt{insert(x)}: inserts \texttt{x}\\[0pt]
\item \texttt{search(k)}: returns the node whose key is \texttt{k}\\[0pt]
\item \texttt{isEmpty()}: returns whether list is empty\\[0pt]
\item \texttt{delete(x)}: given the node \texttt{x}, removes it from the list\\[0pt]
\end{itemize}

    this->next = next;

  bool has_next() {
    return this->next != nullptr;
  }

    Node<T>* n;
    for (n = this; n->get_next() != nullptr; n = n->get_next());
    return n;
  }
};

\subparagraph{LinkedList}

\begin{lstlisting}[language=C++,label=lst:org29d9a55,caption= ,captionpos=b,numbers=none]

template <typename T>
class LinkedList {
private:
  Node<T>* head;

public:
  LinkedList() : head(nullptr) {}
  ~LinkedList() {
    Node<T>* n = this->head;
    while (n) {
      Node<T>* next = n->get_next();
      delete n;
      n = next;
    }
  }
  Node<T>* get_head() {
    return this->head;
  }
  void set_head(Node<T>* n) {
    this->head = n;
  }
  LinkedList* add(int key) {
    return this->add(new Node<T>(key));
  }
  LinkedList* add(Node<T>* n) {
    if (this->is_empty()) {
      this->head = n;
    } else {
      this->get_tail()->set_next(n);
    }
    return this;
  }
  Node<T>* get_tail() {
    if (this->is_empty())
      return nullptr;
    return this->head->get_tail();
  }
  bool is_empty() {
    return this->head == nullptr;
  }
  Node<T>* get_node(int index) {
    Node<T>* n = head;
    for (int i = 0; i < index; ++i) {
      n = n->get_next();
    }
    return n;
  }
  int size() {
    Node<T>* n;
    int i = 0;
    for (n = this->head; n != nullptr; n = n->get_next()) {
      ++i;
    }
    return i;
  }
};

\subsubsection{array list}
\label{sec:orgd284d9b}
\begin{definition}
a \textbf{linked list} is a \hyperref[sec:org39a1299]{concrete data type} that is an implementation of a \hyperref[sec:org250337b]{list} based on an \href{20220728190531-array.org}{array} that is dynamically continuously reconstructed on insertions to hold as many objects as necessary\\[0pt]

\begin{note}
some might suggest that a linked list should shrink automatically aswell on deletions, though that isnt a popular implementation\\[0pt]
\end{note}

\begin{characteristic}
an array list contains a fixed-size array initialized to a constant length, when the array gets filled we create a new array whose size is \(n \times c\) where \(n\) is a constant (usually 2), and \(c\) is the previous size of the array, and we copy all the objects to the new array and then we have more space to store more objects\\[0pt]
\end{characteristic}
\end{definition}
\paragraph{ArrayList}
\label{sec:org705237c}
\begin{lstlisting}[language=C++,label=lst:org6b073e6,caption= ,captionpos=b,numbers=none]
#include <cassert>
#include <memory>

#define INITIAL_SIZE 3
#define SCALING_FACTOR 2

template <typename T> class ArrayList {
private:
  int actual_size;
  int idx;
  T* arr;
  T* allocate_memory(int num_of_objects) {
    this->actual_size = num_of_objects;
    // return new T[num_of_objects];
    // return alloc.allocate(num_of_objects);
    return (T*)::operator new(num_of_objects * sizeof(T));
  }
  void resize() {
    int prev_size = this->actual_size;
    T* newArr = allocate_memory(this->actual_size * SCALING_FACTOR);
    for (int i = 0; i < idx; ++i) {
      newArr[i] = this->arr[i];
    }
    // delete[] this->arr;
    // alloc.deallocate(this->arr, prev_size);
    ::operator delete(this->arr);
    this->arr = newArr;
  }

public:
  ArrayList() {
    this->idx = 0;
    this->arr = this->allocate_memory(INITIAL_SIZE);
  }
  ~ArrayList() {
    delete[] this->arr;
  }
  ArrayList<T>* add(T val) {
    if (this->idx == this->actual_size)
      this->resize();
    this->arr[this->idx++] = val;
    return this;
  }
  T& get(int i) {
    return this->arr[i];
  }
  T& operator [](int i) {
    return this->arr[i];
  }
  void set(int index, T obj) {
    this->arr[index] = obj;
  }
  int size() {
    return this->idx;
  }
  void remove(int index) {
    assert(index < this->idx);
    for (int i = index; i < this->idx-1; ++i) {
      this->arr[i] = this->arr[i+1];
    }
    this->idx--;
  }
  void insert(int index, T value) {
    assert(index < idx);
    if (idx == this->actual_size-1)
      this->resize();
    for (int i = idx; i > index; --i)
      this->arr[i] = this->arr[i-1];
    this->arr[index] = value;
    idx++;
  }
  static ArrayList<T>* from_array(T* other_arr, int length) {
    ArrayList<T>* list = new ArrayList<T>();
    for (int i = 0; i < length; ++i) {
      list->add(other_arr[i]);
    }
    return list;
  }
};
\end{lstlisting}

\subsection{tree}
\label{sec:org6b67a14}
\begin{definition}
a \textbf{tree} is an \hyperref[sec:orga3b31cf]{abstract data type} which stores data in the shape of a downwards-growing tree and every entry is called a \hyperref[sec:orgec4976f]{node}\\[0pt]
\end{definition}
\subsubsection{node}
\label{sec:orgec4976f}
trees store data as \textbf{nodes}, meaning each item in a tree is called a node\\[0pt]
each node \texttt{x} has the following fields\\[0pt]

\begin{itemize}
\item \texttt{key(x)}: a key that's used to identify the node (like every data structure)\\[0pt]
\item \texttt{info(x)}: contains the data we need to store (like every data structure)\\[0pt]
\item \texttt{left(x)}: a pointer to the left child of \texttt{x}\\[0pt]
\item \texttt{right(x)}: a pointer to the right child\\[0pt]
\item \texttt{parent(x)} (optional): a pointer to the parent of \texttt{x}\\[0pt]
\end{itemize}

every node in a \hyperref[sec:org254f5f7]{binary tree} has from 0 to 2 children\\[0pt]
\paragraph{height}
\label{sec:orgb8014c2}
\begin{definition}
the \textbf{height} of a \hyperref[sec:orgec4976f]{node} \texttt{x} is defined as the \hyperref[sec:org2cfcbaa]{height of subtree} of said \hyperref[sec:orgec4976f]{node}, denoted by \(T_x\)\\[0pt]
\end{definition}
\paragraph{balance factor}
\label{sec:org98c9b00}
the \textbf{balance factor} of a \hyperref[sec:orgec4976f]{node} \texttt{x}, denoted by \texttt{balance(x)} or \texttt{BF(x)}, is defined as the height of the \hyperref[sec:orgaaf7b83]{subtree} of the left child of \texttt{x}, denoted by \(T_{\text{left}(x)}\) minus the height of the subtree of the right child of \texttt{x} denoted by \(T_{\text{right}(x)}\), in short:\\[0pt]
\[
  BF(X) = \text{height}(T_{\text{left}(x)}) - \text{height}(T_{\text{right}(x)})

\label{sec:orgcd32ba8}
\begin{definition}
an \textbf{almost complete} binary tree is a binary tree from which 0 or more nodes have been deleted continuously from the right part of the last level of nodes\\[0pt]
\begin{characteristic}
the nodes in the last level in an almost complete binary tree have to appear continuously from left to right\\[0pt]
\end{characteristic}

\begin{characteristic}
every \hyperref[sec:org8d4eeb6]{complete tree} is an \hyperref[sec:orgcd32ba8]{almost complete tree} but not every almost complete tree is a complete tree\\[0pt]
\end{characteristic}

\begin{characteristic}
every level contains 2 times as many nodes as the level before it, except perhaps for the last level\\[0pt]
\end{characteristic}

\begin{characteristic}
the height of a tree is \(\floor*{\log_2 n}\)\\[0pt]
\end{characteristic}

\begin{characteristic}
the number of nodes in a tree is \(\ceil*{\frac{n}{2}}\)\\[0pt]
\end{characteristic}

\begin{characteristic}
the number of nodes at height \(i\) is \(\ceil*{\frac{n}{2i+1}}\)\\[0pt]
\end{characteristic}
\end{definition}
\subsubsection{breadth-first search}
\label{sec:org5127bc8}
\textbf{breadth-first search} or \textbf{BFS} for short, explores each level before moving onto the next one\\[0pt]
extra memory, usually a \hyperref[sec:orga0bf8b2]{queue}, is needed to keep track of the child nodes that were encountered but not yet explored.\\[0pt]

\subsubsection{depth-first search}
\label{sec:org9c80d34}
\textbf{depth-first search} or \textbf{DFS} for short, consists of exploring branches as far as possible before \hyperref[sec:orgee54fe7]{backtracking} and exploring branches of other nodes\\[0pt]
this is the algorithm we'll be using because its simpler\\[0pt]

\paragraph{inorder traversal}
\label{sec:org719f705}
left, parent, right\\[0pt]
\begin{lstlisting}[language=C,label= ,caption= ,captionpos=b,numbers=none]
Algorithm inorder(tree)
  1. go to the left child, i.e., call inorder(left-child), print it
  2. go back to the parent, print it
  3. go to the right child, i.e., call inorder(right-child), print it
\end{lstlisting}
\begin{note}
note that before printing a child will always call \texttt{inorder(left-child)}, so the first node to print would be the leftmost node in the deepest level\\[0pt]
\end{note}
\paragraph{preorder traversal}
\label{sec:org8b23efe}
parent, left, right\\[0pt]
\begin{lstlisting}[language=C,label= ,caption= ,captionpos=b,numbers=none]
Algorithm preorder(tree)
  1. visit the parent, print it
  2. go to the left child, i.e., call preorder(left-child), print it
  3. go to the right child, i.e., call preorder(right-child), print it
\end{lstlisting}
\paragraph{postorder traversal}
\label{sec:orgdf98ea5}
left, right, parent\\[0pt]
\begin{lstlisting}[language=C,label= ,caption= ,captionpos=b,numbers=none]
Algorithm postorder(tree)
  1. go to the left child, i.e., call postorder(left-child), print it
  2. go to the right child, i.e., call postorder(right-child), print it
  3. visit the parent, print it
\end{lstlisting}

bool node_descendants(BinaryTreeNode<int>* n, int k) {
  if (n->is_leaf()) {
    return k == 0;

  if (n->has_right()) new_num--;
  if (n->has_left()) new_num--;
  if (n->has_right() && node_descendants(n->get_right(), new_num))

    return true;
  if (n->has_left() && node_descendants(n->get_left(), k))
    return true;
  return false;
}

\label{sec:org8a19b6e}
a new key is always inserted as a \hyperref[sec:orge114c93]{leaf}. we start from the \hyperref[sec:org76387eb]{root}, making our way downwards by comparing the new key to the key of the current node to decide which child to move to on each iteration, until we hit a leaf node, then we add the new node as its child\\[0pt]

template <typename T> class BinarySearchTree : public BinaryTree<T> {
public:
  BinarySearchTree() {}
  BinarySearchTree(BinaryTreeNode<T>* node) : BinaryTree<T>(node) {}
  virtual BinarySearchTree<T>* insert(BinaryTreeNode<T>* node) {
    node->set_right(nullptr);
    node->set_left(nullptr);
    if (this->is_empty()) {
      BinaryTree<T>::set_root(node);
    } else {
      BinaryTreeNode<T>* n = this->get_root();
      while (true) {
        if (node->get_key() < n->get_key()) {
          if (!n->has_left()) {
            n->set_left(node);
            node->set_parent(n);
            break;
          }
          n = n->get_left();
        } else {
          if (!n->has_right()) {
            n->set_right(node);
            node->set_parent(n);
            break;
          }
          n = n->get_right();
        }
      }
    }
    return this;
  }
  BinarySearchTree<T>* insert(int key, T value) {
    return this->insert(new BinaryTreeNode<T>(key, value));
  }
  BinarySearchTree<T>* insert(int value) {
    return this->insert(new BinaryTreeNode<T>(value));
  }

    for (int i = 0; i < size; ++i)
      this->insert(new BinaryTreeNode<T>(values[i]));
    return this;
  }
  /* this needs to be implemented asap */
  void remove(BinaryTreeNode<T>* node) {
    /* in case that node is the only one in the tree */
    if (this->get_root()->is_leaf() && this->get_root() == node) {
      this->set_root(nullptr);
      return;
    }
    /* in case the node has no children */
    BinaryTreeNode<T>* parent = node->get_parent();
    if (node->is_leaf()) {
      if (node->is_right_child()) {
        parent->set_right(nullptr);
      } else {
        parent->set_left(nullptr);
      }
    }
  }
  void rotate_right(BinaryTreeNode<T>* node) {
    BinaryTreeNode<T>* previous_left = node->get_left();
    if (node->is_left_child()) {
      node->get_parent()->set_left(previous_left);
    } else if (node->is_right_child()) {
      node->get_parent()->set_right(previous_left);
    } else if (node == this->get_root()) {
      this->set_root(previous_left);
      previous_left->set_parent(nullptr);

    node->set_left(previous_left->get_right());

\begin{lemma}
let \texttt{T} be an AVL tree with \texttt{n} nodes and height \(h\), then \(h = \Theta(\log n)\)\\[0pt]
\end{lemma}
\paragraph{minimal form}
\label{sec:orgd4eefed}
\begin{definition}
a minimal form of an \hyperref[sec:orge9da911]{AVL tree} with a given height \(h\) is one which contains the least possible number of nodes\\[0pt]
\begin{lemma}
the minimal number of nodes for a tree of height \(h\) is given by the following \\[0pt]
\begin{align*}
  N(h)&=N(h-1) + N(h-2) + 1\\
  N(1)&=1\\
  N(2)&=2
\end{align*}
\begin{note}
according to some sources, the base cases are \(N(0)=1,\ N(1)=2\), but the base cases given above make more sense because a tree with a single node should have a height of 1 not 0\\[0pt]
\end{note}
\end{lemma}

\begin{lemma}
a minimal form of an \hyperref[sec:orge9da911]{AVL tree} with a given number of nodes \(N\) is the one with the smallest height which is given by:\\[0pt]
\[
  h = \log_2 N
\]\\[0pt]
\end{lemma}

\paragraph{insertion}
\label{sec:orgf7c2dfc}
the process of inserting a node to an AVL tree starts with inserting it using the same insertion method we used for a \hyperref[sec:orge1ff544]{binary search tree}, after said insertion we start going up the tree starting at the new node we just inserted and updating the \hyperref[sec:org98c9b00]{balance factor} for every node we encounter as we go\\[0pt]
if we arrive at \hyperref[sec:org634a573]{problematic node}, we apply the proper \hyperref[sec:orgcca3d84]{rotation} to \hyperref[sec:org224f44b]{rebalance} the tree\\[0pt]
\paragraph{rebalancing}
\label{sec:org224f44b}
during a modifying operation of a \hyperref[sec:org254f5f7]{binary tree}, a node might become \hyperref[sec:org634a573]{problematic}, to restore balance to the tree so that it remains an \hyperref[sec:orge9da911]{AVL tree} we use \hyperref[sec:orgcca3d84]{rotation}s\\[0pt]
\subparagraph{rotation}
\label{sec:orgcca3d84}
to determine which type of \textbf{rotation} we need to use to \textbf{rebalance} a given tree, we go 2 steps from the \hyperref[sec:org634a573]{problematic node} towards the newly inserted node which gives us the following 4 cases:\\[0pt]
\begin{enumerate}
\item if we go 2 times to the left, then the \textbf{rotation} is of type \hyperref[sec:orga4dd83f]{LL}\\[0pt]
\item if we go 2 times to the right, then the \textbf{rotation} is of type \hyperref[sec:orgae1b93d]{RR}\\[0pt]
\item if we go left and then right then the \textbf{rotation} is of type \hyperref[sec:org9aa5be8]{LR}\\[0pt]
\item if we go right and then left then the \textbf{rotation} is of type \hyperref[sec:org56c5343]{RL}\\[0pt]
\end{enumerate}
\begin{enumerate}
\item simple rotation
\label{sec:orga681800}
\begin{enumerate}
\item left left rotation
\label{sec:orga4dd83f}
let \texttt{z} be the left child of the \hyperref[sec:org634a573]{problematic node} and \texttt{x} be the problematic node itself\\[0pt]
in \textbf{Left Left} rotation we rotate \texttt{x} to the right, such that the left child of \texttt{x} replaces \texttt{x} and \texttt{x} gets "rotated" to the right becoming the right child of \texttt{z} and a parent of the previous right child of \texttt{z}\\[0pt]

template <typename T> class AVLTree : public BinarySearchTree<T> {
public:
  using BinarySearchTree<T>::insert;
  AVLTree() {}
  AVLTree(BinaryTreeNode<T>* node) : BinarySearchTree<T>(node) {}
  AVLTree<T>* insert(BinaryTreeNode<T>* node) override {
    BinarySearchTree<T>::insert(node);
    BinaryTreeNode<T>* retracing_node = node;
    while (retracing_node != nullptr) {
      if (retracing_node->is_problematic()) {
        BinaryTreeNode<T>* path_root = retracing_node->find_path(node);
        if (path_root->has_left() && path_root->get_left()->has_left()) { /* LL rotation */
          this->rotate_right(retracing_node);
        }
        if (path_root->has_right() && path_root->get_right()->has_right()) { /* RR rotation */
          this->rotate_left(retracing_node);
        }
        if (path_root->has_right() && path_root->get_right()->has_left()) { /* RL rotation */
          this->rotate_right(retracing_node->get_right()); /* LL rotation */
          this->rotate_left(retracing_node); /* RR rotation */
        }
        if (path_root->has_left() && path_root->get_left()->has_right()) { /* LR rotation */
          this->rotate_left(retracing_node->get_left()); /* RR rotation */
          this->rotate_right(retracing_node); /* LL rotation */
        }
          delete path_root;
          break;
        }
      retracing_node = retracing_node->get_parent();
    }
    return this;
  }
};

\subsubsection{2-3 tree}
\label{sec:org1df410f}
a \textbf{2–3 tree} is a \hyperref[sec:org6b67a14]{tree} \hyperref[sec:org39a1299]{concrete data type} where every \hyperref[sec:org26c1a68]{internal node} is either a \hyperref[sec:orgcb7e48c]{2-node} or \hyperref[sec:org08b0689]{3-node}\\[0pt]
an exceptional case is when the tree contains only one element in which case the only node in the tree (which is the \hyperref[sec:org76387eb]{root}) holds only one element\\[0pt]
\begin{characteristic}
every \hyperref[sec:org26c1a68]{internal node} is a \hyperref[sec:orgcb7e48c]{2-node} or a \hyperref[sec:org08b0689]{3-node} \\[0pt]
\end{characteristic}
\begin{characteristic}
all \hyperref[sec:orge114c93]{leaves} are at the same level\\[0pt]
\end{characteristic}
\begin{characteristic}
keys are always ordered from left to right in ascending order\\[0pt]
\end{characteristic}
\paragraph{2-node}
\label{sec:orgcb7e48c}
\begin{definition}
we say that an \hyperref[sec:org26c1a68]{internal node} is a \textbf{2-node} if it holds one value and two children\\[0pt]
\end{definition}
\paragraph{3-node}
\label{sec:org08b0689}
\begin{definition}
we say that an \hyperref[sec:org26c1a68]{internal node} is a \textbf{3-node} if it holds two values and three children\\[0pt]
\end{definition}
\paragraph{insertion}
\label{sec:orgc48a523}
an important property to keep in mind is that insertion \textbf{always} occurs in one of the \hyperref[sec:orge114c93]{leaves}\\[0pt]

        this->children[i] = nullptr;
        return;
      }

        return i;

\subsubsection{algorithms}
\label{sec:orgf60ac12}
\paragraph{backtracking}
\label{sec:orgee54fe7}
\textbf{Backtracking} is an algorithm that incrementally builds candidates to the solutions, and abandons a candidate as soon as it determines that the candidate cannot possibly be completed to a valid solution, see \url{https://en.wikipedia.org/wiki/Backtracking}\\[0pt]
after solving a few problems i saw a pattern that makes it much easier to solve recursion problems related to binary trees\\[0pt]
when i want to solve a problem with recursion i ask myself 3 questions\\[0pt]
\begin{enumerate}
\item what is a node's job?\\[0pt]
\item what info does a node require to do its job?\\[0pt]
\item how do we give the node what it requires to do its job? (one of 2 methods)\\[0pt]
\end{enumerate}
the 2 methods of passing data around binary tree nodes are:\\[0pt]
\begin{itemize}
\item if a node needs to know some info about its \hyperref[sec:org48e7ebc]{ancestor}s or nodes from other subtrees this info has to be passed through function arguments because once we reach a node in the tree we cant go back to previous nodes (its possible but would take \texttt{O(n)} time which is horrible)\\[0pt]
\item if a node needs to know something about its \hyperref[sec:org3e6c055]{descendant}s, this information has to be backpropagated by returning it to the parent node that called the recursive function on its child, because once we're back to a node we cant go back down (we could but again, it takes \texttt{O(n)} time), if we need to backpropagate multiple values we can use a simple structure to hold the values (in \texttt{C++} the simplest way is to use \texttt{struct})\\[0pt]
\end{itemize}
note that its \textbf{important} that we gather info as we go on every node we visit so that once we leave a specific node we'd have all the info we would need from it so that we wouldnt have to go back to visit it which usually results in \texttt{O(n)} time\\[0pt]
words arent the best way to put it so hopefully ill provide a better visual explanation in the future\\[0pt]
\paragraph{reverse inorder}
\label{sec:org45f4751}
given a \hyperref[sec:org254f5f7]{binary tree}, write a function to reverse its \hyperref[sec:org719f705]{inorder traversal} output\\[0pt]
this algorithm runs in \texttt{O(n)} time\\[0pt]
\begin{lstlisting}[language=C++,label=lst:orge5cfdea,caption= ,captionpos=b,numbers=none]
template <typename T>
void reverse(BinaryTreeNode<T>* node) {
  if (node == nullptr)
    return;
  reverse(node->get_left());
  reverse(node->get_right());
  BinaryTreeNode<T>* l = node->get_left();
  node->set_left(node->get_right());
  node->set_right(l);
}
\end{lstlisting}

\subsection{heap}
\label{sec:org9903735}
\begin{definition}
a \textbf{Heap} is an \hyperref[sec:orga3b31cf]{abstract data type}\\[0pt]
\end{definition}
\subsubsection{min heap}
\label{sec:orgc87ebb0}
\begin{definition}
a \textbf{min heap} is a \hyperref[sec:org9903735]{heap} in which each \hyperref[sec:orgec4976f]{node} is less than or equal to both its children\\[0pt]
\end{definition}
\subsubsection{max heap}
\label{sec:org75e2e7b}
\begin{definition}
a \textbf{max heap} is a \hyperref[sec:org9903735]{heap} in which each node is bigger than or equal to both its children\\[0pt]
\end{definition}
this is the implementation we'll be using\\[0pt]
\subsubsection{binary heap}
\label{sec:orgff15739}
\begin{definition}
A \textbf{binary heap} is defined as a \hyperref[sec:org254f5f7]{binary tree} with two additional constraints:\\[0pt]
\begin{enumerate}
\item shape property: a binary heap is an \hyperref[sec:orgcd32ba8]{almost complete tree}\\[0pt]
\item heap property: the key stored in each node is either greater than or equal to, or less than or equal to the keys in the node's children\\[0pt]
\end{enumerate}

\begin{characteristic}
a heap may keep a pointer to speed up insertion, this pointer points to the leftmost leaf if the tree is complete, otherwise if its almost complete, it points to the right uncle of the rightmost leaf\\[0pt]
\end{characteristic}
\begin{note}
in a heap the keys arent necessarily sorted like in \hyperref[sec:orge1ff544]{binary search tree}s\\[0pt]
\end{note}
\end{definition}
\paragraph{bubbling}
\label{sec:org74642bd}
\begin{definition}
moving a \hyperref[sec:orgec4976f]{node} up/down a heap to get it to its correct position\\[0pt]
\end{definition}
\subparagraph{bubble up}
\label{sec:org389e434}
\hyperref[sec:org74642bd]{bubbling} a node \textbf{upwards} consists of recursively switching it with its parent until its parent is bigger/smaller than it (depending on whether we're using a \hyperref[sec:orgc87ebb0]{min heap} or a \hyperref[sec:org75e2e7b]{max heap})\\[0pt]
\subparagraph{bubble down}
\label{sec:org3bbfb76}
\hyperref[sec:org74642bd]{bubbling} a node \textbf{downards} consists of recursively switching it with one of its children until it gets to its correct position\\[0pt]
this is a pseudocode for the \href{20220706211958-algorithm.org}{algorithm}:\\[0pt]
\begin{lstlisting}[language=C,label= ,caption= ,captionpos=b,numbers=none]
l = left(i) // index of left child
r = right(i) // index of right child
if l <= heap-size[A] and A[l] > A[i]
  largest = l
else
  largest = i
if r <= heap-size[A] and A[r] > A[largest]
  largest = r
if largest != i
  exchange A[i] A[largest]
  heapify(A, largest)
\end{lstlisting}
\href{20221130014441-time_complexity.org}{time complexity} is \(O(\log n)\)\\[0pt]
\paragraph{insertion}
\label{sec:org2f88fcf}
inserting a new \hyperref[sec:orgec4976f]{node} to the heap consists of adding it as a \hyperref[sec:orge114c93]{leaf} such that the heap remains \hyperref[sec:orgcd32ba8]{almost complete} and \hyperref[sec:org389e434]{bubbling up} the newly added node so that the keys in the tree remain sorted\\[0pt]
if the heap is in \hyperref[sec:org93e8c49]{array representation}, we insert the new node to the end of the array and heapify it upwards (bubble up)\\[0pt]
\paragraph{deletion}
\label{sec:org40f555e}
deletion of a node consists of replacing the rightmost leaf with it and bubble it \hyperref[sec:org3bbfb76]{downwards} if its smaller than one of its children or \hyperref[sec:org389e434]{upwards} if its bigger than its parent\\[0pt]
\paragraph{binary heap array representation}
\label{sec:org93e8c49}
\begin{note}
some resources may use assume that arrays start at index 1, i see no point in that since most programming languages use 0 as the first index, so some of the following propositions might seem a little different like for example the proposition number 2 in the following list according to these sources would be that the left child of i\textsuperscript{th} node is \(Arr[2i]\) instead of \(Arr[2i+1]\) and the right child is \(Arr[2i+1]\)\\[0pt]
\end{note}
for an array to represent a \hyperref[sec:orgff15739]{binary heap}, the only condition is that \(A[i] < A\left[\floor*{\frac{i-1}{2}}\right]\) (in the case of \hyperref[sec:org75e2e7b]{max heap})\\[0pt]
arrays that represent heap have the following characterstics:\\[0pt]
\begin{enumerate}
\item root is \(Arr[0]\)\\[0pt]
\item the parent of the i\textsuperscript{th} node (in the array) is \(A\left[\floor*{\frac{i-1}{2}}\right]\)\\[0pt]
\item left child of the i\textsuperscript{th} node is \(Arr[2i+1]\)\\[0pt]
\item right child of the i\textsuperscript{th} node is \(Arr[2i+2]\)\\[0pt]
\end{enumerate}
in other words, the nodes are taken from the root to the last level from left to right and put in that order in an array\\[0pt]
the array should hold 2 values, \texttt{array-size}, which is the size of the array, and \texttt{heap-size}, which is the number of elements of the heap in the array\\[0pt]
\paragraph{construction of binary heap from array}
\label{sec:org8e8c688}
given some random \href{20220728190531-array.org}{array}, e.g. \texttt{A=[10,20,30,40,50,70,35,60,80]}, we need to reorganize its elements such that it \hyperref[sec:org93e8c49]{represents a binary heap}\\[0pt]
to achieve this we apply \hyperref[sec:org3bbfb76]{bubble down} to the first half of the array \textbf{in reverse} (because the \hyperref[sec:orge114c93]{leaf} nodes need not be heapified as they already follow the heap property)\\[0pt]
e.g. for the given array we apply \texttt{bubble(A, i)} such that \(i = 3 \to 0\)\\[0pt]
this is the constructing \href{20220706211958-algorithm.org}{algorithm} in pseudocode:\\[0pt]
\begin{lstlisting}[language=C,label= ,caption= ,captionpos=b,numbers=none]
Build-Heap(A)
  n = length(A)
  heap-size = n
  for i = n/2 to 1
    heapify(A, i)
\end{lstlisting}
\begin{lemma}
this algorithm runs in \(\Theta(n)\) time\\[0pt]
\begin{proof}
for \(0 < k < 1\) we know:\\[0pt]
\[
  \sum_{x=1}^{\infty} x \cdot k^x = \frac{k}{(1-k)^2}
\]\\[0pt]
so for \(k=\frac12\):\\[0pt]
\[
  \sum_{x=1}^{\infty} x \cdot \left(\frac{1}{2}\right)^x = \frac{1}{2} + \frac{2}{2^2} + \frac{3}{2^3} + \frac{4}{2^4} + \cdots = \frac{\frac{1}{2}}{\left(1-\frac{1}{2}\right)^2} = \frac{\frac{1}{2}}{\frac{1}{4}} = 2
\]\\[0pt]
to be continued\\[0pt]
\end{proof}
\end{lemma}
\paragraph{BinaryHeap}
\label{sec:org2eac639}
\begin{note}
this is a \hyperref[sec:org75e2e7b]{max heap} implementation\\[0pt]
\end{note}
\begin{lstlisting}[language=C++,label=lst:org280ee3b,caption= ,captionpos=b,numbers=none]
#include <cmath>




#define PARENT_INDEX(node_idx)      floor((node_idx-1)/2)
#define LEFT_CHILD_INDEX(node_idx)  2*node_idx+1
#define RIGHT_CHILD_INDEX(node_idx) 2*node_idx+2

template <typename T>
class BinaryHeap {
private:
  ArrayList<KeyValue<int, T>>* list = new ArrayList<KeyValue<int, T>>();
  // heapify downwards assuming max heap
  void heapify_down(int idx) {
    KeyValue<int, T>* current = &this->list->get(idx);
    KeyValue<int, T>* left = LEFT_CHILD_INDEX(idx) >= list->size() ? nullptr : &this->list->get(LEFT_CHILD_INDEX(idx));
    KeyValue<int, T>* right = RIGHT_CHILD_INDEX(idx) >= list->size() ? nullptr : &this->list->get(RIGHT_CHILD_INDEX(idx));
    if (left == nullptr && right == nullptr)
      return;
    if (left == nullptr) {
      if (right->get_key() > current->get_key()) {
        KeyValue<int, T> temp = *current;
        this->list->set(idx, *right);
        this->list->set(RIGHT_CHILD_INDEX(idx), temp);
        heapify_down(RIGHT_CHILD_INDEX(idx));
      }
    } else if (right == nullptr) {
      if (left->get_key() > current->get_key()) {
        KeyValue<int, T> temp = *current;
        this->list->set(idx, *left);
        this->list->set(LEFT_CHILD_INDEX(idx), temp);
        heapify_down(LEFT_CHILD_INDEX(idx));
      }
    } else {
      if (left->get_key() > current->get_key() && left->get_key() > right->get_key()) {
        KeyValue<int, T> temp = *current;
        this->list->set(idx, *left);
        this->list->set(LEFT_CHILD_INDEX(idx), temp);
        heapify_down(LEFT_CHILD_INDEX(idx));
      }
      if (right->get_key() > current->get_key() && right->get_key() > left->get_key()) {
        KeyValue<int, T> temp = *current;
        this->list->set(idx, *right);
        this->list->set(RIGHT_CHILD_INDEX(idx), temp);
        heapify_down(RIGHT_CHILD_INDEX(idx));
      }
    }
  }
  void heapify_up(int idx) {
    if (idx == 0)
      return;
    KeyValue<int, T>* current = &this->list->get(idx);
    KeyValue<int, T>* parent = &this->list->get(PARENT_INDEX(idx));
    if (current->get_key() > parent->get_key()) {
      KeyValue<int, T> temp = *current;
      this->list->set(idx, *parent);
      this->list->set(PARENT_INDEX(idx), temp);
      heapify_up(PARENT_INDEX(idx));
    }
  }
  BinaryTreeNode<T>* to_binary_tree(int idx) {
    if (idx >= this->list->size())
      return nullptr;
    BinaryTreeNode<T>* node = new BinaryTreeNode<T>(this->list->get(idx).get_key(), this->list->get(idx).get_value());
    BinaryTreeNode<T>* left = to_binary_tree(LEFT_CHILD_INDEX(idx));
    BinaryTreeNode<T>* right = to_binary_tree(RIGHT_CHILD_INDEX(idx));
    node->set_left(left);
    node->set_right(right);
    return node;
  }

public:
  ~BinaryHeap() {
    delete this->list;
  }
  BinaryHeap<T>* insert(KeyValue<int, T> kv) {
    this->list->add(kv);
    this->heapify_up(this->list->size()-1);
    return this;
  }
  BinaryHeap<T>* insert(int key) {
    return this->insert(KeyValue<int, int>(key, key));
  }
  KeyValue<int, T>& operator [](int i) {
    return (*list)[i];
  }
  KeyValue<int, T>& get(int i) {
    return (*list)[i];
  }
  static BinaryHeap<T>* from_array(KeyValue<int, T>* other_arr, int length) {
    BinaryHeap<T>* heap = new BinaryHeap<T>();
    heap->list = ArrayList<KeyValue<int, T>>::from_array(other_arr, length);
    for (int i = length / 2; i >= 0; --i) {
      heap->heapify_down(i);
    }
    return heap;
  }
  int size() {
    return this->list->size();
  }
  BinaryTree<T>* to_binary_tree() {
    return new BinaryTree<T>(to_binary_tree(0));
  }
};
\end{lstlisting}

\subsubsection{binomial heap}
\label{sec:orgd6615d4}
some info taken from \url{http://staff.ustc.edu.cn/\~csli/graduate/algorithms/book6/chap20.htm}\\[0pt]
\begin{definition}
a \textbf{binomial heap} is a \hyperref[sec:org39a1299]{concrete data type} that implements the \hyperref[sec:org9903735]{heap} using a set of \hyperref[sec:org6e9f2b7]{binomial tree}s\\[0pt]

\begin{characteristic}
\hyperref[sec:orgff15739]{binary heap}s, binomial heaps, and fibonacci heaps are all inefficient in their support of the operation \texttt{SEARCH}; it can take a while to find a node with a given key. for this reason, operations such as \texttt{DECREASE-KEY} and \texttt{DELETE} that refer to a given node require a pointer to that node as part of their input. this requirement poses no problem in many applications.\\[0pt]
\end{characteristic}
\end{definition}
\paragraph{binomial tree}
\label{sec:org6e9f2b7}
\begin{definition}
a \textbf{binomial tree} is defined recursively as follows:\\[0pt]
\begin{itemize}
\item a binomial tree of order 0 is a single node\\[0pt]
\item a binomial tree of order \(k\) has a root node whose children are roots of binomial trees of orders, \(k-1,k-2,\ldots,2,1,0\) (in this order)\\[0pt]
\end{itemize}

\begin{characteristic}
each binomial tree in a heap obeys the \hyperref[sec:orgc87ebb0]{min heap} property\\[0pt]
this ensures that the root of each binomial tree contains the smallest key in the tree. it follows that the smallest key in the entire heap is one of the roots.\\[0pt]
\end{characteristic}

\begin{characteristic}
there can be at most one binomial tree for each order, including order 0\\[0pt]
this property implies that a binomial heap with \(n\) nodes consists of at most \(1+\log_2n\) binomial trees. the number and orders of these trees are uniquely determined by the number of nodes \(n\): there is one binomial tree for each nonzero bit in the binary representation of the number \(n\). for example, the decimal number 13 is 1101 in binary, \(2^3+2^2+2^0\), and thus a binomial heap with 13 nodes will consist of three binomial trees of orders 3, 2, and 0 (see the \hyperref[orgd409229]{figure below}).\\[0pt]
\end{characteristic}

\begin{characteristic}
a binomial tree of order \(k\) has \(2^{k}\) nodes and height \(k\). the name comes from the shape: a binomial tree of order k has \(\binom{k}{d}\) nodes at depth \(d\), a \href{20221204105120-combinatorics.org}{binomial coefficient}. because of its structure, a binomial tree of order \(k\) can be constructed from two trees of order \(k-1\) by attaching one of them as the leftmost child of the root of the other tree. This feature is central to the \hyperref[sec:org6a4cc43]{merge} operation of a binomial heap, which is its major advantage over other conventional heaps.\\[0pt]
\end{characteristic}
\end{definition}
\paragraph{implementation}
\label{sec:orge873736}
because no operation requires random access to the root nodes of the binomial trees, the roots of the binomial trees can be stored in a \hyperref[sec:org3b67cf5]{linked list}, ordered by increasing order of the tree. because the number of children for each node is variable, it does not work well for each node to have separate links to each of its children, as would be common in a \hyperref[sec:org254f5f7]{binary tree}; instead, it is possible to implement this tree using links from each node to its highest-order child in the tree, and to its sibling of the next smaller order than it. these sibling pointers can be interpreted as the next pointers in a linked list of the children of each node, but with the opposite order from the linked list of roots: from largest to smallest order, rather than vice versa. this representation allows two trees of the same order to be linked together, making a tree of the next larger order, in constant time.\\[0pt]
each node \texttt{x} has a key field and contains pointers \texttt{parent(x)} to its parent, \texttt{child(x)} to its leftmost child, and \texttt{sibling(x)} to the sibling of x immediately to its right. if node x is a root, then \texttt{parent(x) = nil}. if node x has no children, then \texttt{child(x) = nil}, and if x is the rightmost child of its parent, then \texttt{sibling(x) = nil}. each node x also contains the field \texttt{degree(x)}, which is the number of children of x.\\[0pt]
the roots of the binomial trees within a binomial heap are organized in a linked list, which we refer to as the \textbf{root list}. the degrees of the roots strictly increase as we traverse the root list. in an n-node binomial heap the degrees of the roots are a subset of \({0,1,\ldots,\lg n}\). the sibling field has a different meaning for roots than for nonroots. if x is a root, then \texttt{sibling(x)} points to the next root in the root list. (as usual, \texttt{sibling(x) = nil} if x is the last root in the root list.)\\[0pt]
a given binomial heap \texttt{h} is accessed by the field \texttt{head(h)}, which is simply a pointer to the first root in the root list of h. if binomial heap H has no elements, then \texttt{head[h] = nil}.\\[0pt]
\subparagraph{create}
\label{sec:orgb19781c}
to make an empty binomial heap, the \texttt{MAKE-BINOMIAL-HEAP} procedure simply allocates and returns an object \texttt{h}, where \texttt{head[h] = nil}. the running time is \(O(1)\).\\[0pt]
\subparagraph{union}
\label{sec:org6a4cc43}
the operation of uniting two binomial heaps is used as a subroutine by most of the remaining operations. merging of binomial trees is done by comparing the keys at the roots of two trees, and the root node with the larger key will become the child of the root with the smaller key. the time complexity for finding a union is \(O(\log n)\).\\[0pt]
to perform the union of two binomial heaps, first we \textbf{merge} the heaps into one, such that the trees in the resulting heap would be in monotonically ascending order, then we traverse through the roots of the binomial trees, we consider the following cases where \texttt{x} denotes the current root:\\[0pt]
\begin{enumerate}
\item if \texttt{degree(x)} is not equal to \texttt{degree(next(x))}, then move the pointer ahead\\[0pt]
\item if \texttt{degree(x) = degree(next(x)) = degree(next(next(x)))} then move the pointer ahead\\[0pt]
\item if \texttt{degree(x) = degree(next(x))} but not equal to \texttt{degree(next(next(x)))}, and \texttt{key(x) < key(next(x))} then remove \texttt{next(x)} from root and attach it to x\\[0pt]
\item if \texttt{degree(x) = degree(next(x))} but not equal to \texttt{degree(next(next(x)))} and \texttt{key(x) > key(next(x))} then remove x from root and attach it to \texttt{next(x)}.\\[0pt]
\end{enumerate}
because each binomial tree in a binomial heap corresponds to a bit in the binary representation of its size, there is an analogy between the merging of two heaps and the binary addition of the sizes of the two heaps, from right-to-left. whenever a carry occurs during addition, this corresponds to a merging of two binomial trees during the merge.\\[0pt]
each binomial tree's traversal during merge only involves roots, hence making the time taken at the biggest order \(\log_2n\) and therefore the running time is \(O(\log n)\).\\[0pt]
\subparagraph{insert}
\label{sec:org27eb891}
inserting a new element to a heap can be done by creating a new heap containing only this element and then \hyperref[sec:org6a4cc43]{merging} it with the original heap. because of the merge, a single insertion takes time \(O(\log n)\). however, this can be sped up using a merge procedure that shortcuts the merge after it reaches a point where only one of the merged heaps has trees of larger order. with this speedup, across a series of \(k\) consecutive insertions, the total time for the insertions is \(O(k+\log n)\). another way of stating this is that (after logarithmic overhead for the first insertion in a sequence) each successive insert has an amortized time of \(O(1)\) (i.e. constant time) per insertion.\\[0pt]
a variant of the binomial heap, the skew binomial heap, achieves constant worst case insertion time by using forests whose tree sizes are based on the skew binary number system rather than on the binary number system.\\[0pt]

template <typename T>
class BinomialHeap {
private:
  BinomialHeapNode<T>* head = nullptr;
public:
  BinomialHeap() = default;
  BinomialHeap(BinomialHeapNode<T>* head) {
    this->head = head;
  }
  int order() {
    BinomialHeapNode<T>* tail = head->get_tail();
    return tail->degree();
  }
  bool is_empty() {
    return this->head == nullptr;
  }
  BinomialHeap<T>* insert(BinomialHeapNode<T>* node) {
    if (this->is_empty())
      this->head = node;
    else
      this->unite(new BinomialHeap<T>(node));
    return this;
  }
  BinomialHeapNode<T>* get_head() {
    return this->head;
  }
  void merge(BinomialHeap<T>* other) {
    BinomialHeapNode<T>* this_iterator = this->head;
    BinomialHeapNode<T>* n = other->head;
    while (n != nullptr) {
      while (this_iterator->has_next() && ((BinomialHeapNode<T>*)this_iterator->get_next())->degree() < n->degree())
        this_iterator = (BinomialHeapNode<T>*)this_iterator->get_next();
      BinomialHeapNode<T>* old_n_next = (BinomialHeapNode<T>*)n->get_next();
      if (n->degree() < this->head->degree()) {
        n->set_next(this->head);
        this->head = n;
        this->head->set_parent(nullptr); /* TODO should reset child and parent */
      } else {
        BinomialHeapNode<T>* old_this_next = (BinomialHeapNode<T>*)this_iterator->get_next();
        this_iterator->set_next(n);
        n->set_next(old_this_next);
      }
      n = old_n_next;
    }
  }
  void unite(BinomialHeap<T>* other) {
    this->merge(other);
    BinomialHeapNode<T>* n = this->head;
    BinomialHeapNode<T>* prev = nullptr;
    while (n != nullptr && n->has_next()) {
      BinomialHeapNode<T>* next = (BinomialHeapNode<T>*)n->get_next();
      BinomialHeapNode<T>* nextnext = n->has_next() ? (BinomialHeapNode<T>*)n->get_next()->get_next() : nullptr;
      bool next_same = next != nullptr && n->degree() == next->degree();
      bool nextnext_same = nextnext != nullptr && n->degree() == nextnext->degree();
      if (!next_same) {
        prev = n;
        n = (BinomialHeapNode<T>*)n->get_next();
        // std::cout << "case 1" << std::endl;
      } else if (next_same && nextnext_same) {
        prev = n;
        n = (BinomialHeapNode<T>*)n->get_next();
        // std::cout << "case 2" << std::endl;
      } else if (next_same && !nextnext_same && n->get_key() < next->get_key()) {
        // remove next and attach it to n
        // std::cout << "attaching " << next->get_key() << " to " << n->get_key() << std::endl;
        next->set_next(n->get_child());
        n->set_next(nextnext);
        n->set_child(next);
        // std::cout << "case 3" << std::endl;
      } else if (next_same && !nextnext_same && n->get_key() > next->get_key()) {
        // remove n and attach it to next
        // std::cout << "attaching " << n->get_key() << " to " << next->get_key() << std::endl;
        BinomialHeapNode<T>* new_n = next;
        n->set_next(next->get_child());
        if (prev != nullptr) {
          prev->set_next(next);
        } else { /* its the first root now */
          this->head = next;
        }
        next->set_child(n);
        // std::cout << "case 4" << std::endl;
        n = new_n;
      }
    }
  }
};

\subsection{hash table}
\label{sec:org4f64d21}
\begin{definition}
a \textbf{hash table}, is an \hyperref[sec:orga3b31cf]{abstract data type} that maps keys to values. a hash table uses a hash function to compute an index into an array of buckets or slots, from which the desired value can be found. during lookup, the key is hashed and the resulting hash indicates where the corresponding value is stored.\\[0pt]
it contains the following functions:\\[0pt]
\begin{itemize}
\item \texttt{Insert(A, x)}\\[0pt]
\item \texttt{Delete(A, x)}\\[0pt]
\item \texttt{Search(A, x)}\\[0pt]
\end{itemize}
\end{definition}
\subsection{theorems}
\label{sec:org97a5c46}
\begin{lemma}
the number of \hyperref[sec:orgec4976f]{node}s in a \hyperref[sec:org8d4eeb6]{complete tree} of height \(h\) is \(n = 2^h-1\)\\[0pt]
\begin{proof}
we add all the nodes in a given complete tree according to their levels to get \(n = 1+2+2^2+\cdots+2^{h-1}=2^h-1\)\\[0pt]
\end{proof}
\end{lemma}

\begin{lemma}
the height of a complete tree with \(n\) nodes is \(\log_2(n+1)\)\\[0pt]
\begin{proof}
according to the previous lemma \(n+1 = 2^h\), therefore \(h = \log_2(n+1)\)\\[0pt]
\end{proof}
\end{lemma}

\begin{lemma}
the number of \hyperref[sec:orge114c93]{leaves} in a complete tree with \(n\) nodes and height \(h\) is \(2^{h-1}\)\\[0pt]
\begin{proof}
using \href{20220707193301-mathematical_induction.org}{induction}\\[0pt]
the base case is \(h=1\), a complete tree of height 1 contains 1 node only which is a leaf, so \(2^{h-1}=2^{1-1}=2^0=1\)\\[0pt]
we assume the statement holds true for every \(h \leq i\)\\[0pt]
we prove the statements holds true for \(h=i+1\), looking at a complete tree of height \(i+1\), we can build it using a tree of height \(i\) by adding 2 children to every leaf, so \(2 \cdot 2^{i-1} = 2^i\) which means the statement holds true for \(i+1\)\\[0pt]
\end{proof}
\end{lemma}

\begin{lemma}
the number of leaves in a complete tree that holds \(n\) nodes is \(\frac{n+1}{2}\)\\[0pt]
\begin{proof}
we denote by \(h\) the height of the tree:\\[0pt]
\[
  \text{number of leaves} = 2^{h-1} = 2^{\log_2(n+1)-1} = \frac{2^{\log_2(n+1)}}{2} = \frac{n+1}{2}
\]\\[0pt]
\end{proof}
\end{lemma}

\begin{lemma}
in an \hyperref[sec:orgcd32ba8]{almost complete tree}, the number of nodes in a specific level is 2 times the number of nodes in level before it (except for perhaps the last level)\\[0pt]
\end{lemma}

\begin{lemma}
the height of an almost complete tree with \(n\) nodes is \(\floor*{\log_2 n}\)\\[0pt]
\end{lemma}

\begin{lemma}
the number of leaves in an almost complete tree with \(n\) nodes is \(\ceil*{\frac{n}{2}}\)\\[0pt]
\end{lemma}

\begin{lemma}
the number of nodes with height \(i\) in an almost complete tree with \(n\) nodes is \(\ceil*{\frac{n}{2^i+1}}\)\\[0pt]
\end{lemma}

\begin{lemma}
for every almost \hyperref[sec:orgcd32ba8]{almost complete tree}complete tree with \(n\) nodes and height \(h\), \(2^{h-1} \leq n \leq 2^h-1\)\\[0pt]
\begin{proof}
the biggest almost complete tree of height \(h\) is a complete tree and so \(n \leq \text{nodes in complete tree of height h} = 2^h-1\)\\[0pt]
the smallest almost complete tree of height \(h\) can be constructed by adding a left child to the leftmost leaf of a complete tree of height \(h-1\), \(n \geq \text{number of nodes in tree of height h-1 plus one} = 2^{h-1}-1+1=2^{h-1}\)\\[0pt]
\end{proof}
\end{lemma}

\begin{lemma}
for every almost complete tree with \(n\) nodes and height \(h\), \(\log_2(n+1) \leq h \leq 1+\log_2{n}\)\\[0pt]
\end{lemma}

\begin{lemma}
for every almost complete tree with \(n\) nodes and height \(h\), \(2^{h-2} \leq \text{number of leaves} \leq 2^{h-1}\)\\[0pt]
\end{lemma}
