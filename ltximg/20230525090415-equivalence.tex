\begin{definition}
\href{20230401175139-statement.org}{statement}s \(p\) and \(q\) are said to be \textbf{logically equivalent} if they have the same truth value in every model.\\
\end{definition}
proving equivalence can be done:\\
\begin{itemize}
\item using a \href{20230116122956-truth_table.org}{truth table} and verifying that all rows are similar\\
\item using already proven equivalences\\
\item using an \href{20230525071323-formal_proof.org}{inference process}\\
\end{itemize}
\begin{my_example}
proving \(\lnot (A \lor B) \land B \lor D \equiv D\)\\
\begin{proof}
\begin{align*}
  \lnot (A \lor B) \land B \lor D &\equiv (\lnot A \land \lnot B) \land B \lor D & E16\\
  &\equiv \lnot A \land (\lnot B \land B) \lor D & E13\\
  &\equiv \lnot A \land F \lor D & E8\\
  &\equiv (\lnot A \land F) \lor D\\
  &\equiv F \lor D & E4\\
  &\equiv D & E2
\end{align*}
\end{proof}
\end{my_example}
