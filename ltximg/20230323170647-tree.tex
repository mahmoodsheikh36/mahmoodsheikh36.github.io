\begin{definition}
a connected, non-directed, acyclic \href{20230314001051-graph.org}{graph} is called a \textbf{tree}\\
\end{definition}
\begin{lemma}
the following cases are equal:\\
\begin{enumerate}
\item the graph \(G\) is a tree\\
\item between every two vertices there is a single path\\
\item \(G\) is connected and upon the removal of any of the edges we get a non-connected graph\\
\item \(G\) is connected and \(|E|=|V|-1\)\\
\item \(G\) is acyclic and \(|E|=|V|-1\)\\
\end{enumerate}
\end{lemma}
